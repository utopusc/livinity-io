Wrapping pattern example (FadeIn over in-view):
```tsx
// livos/packages/ui/src/components/motion/FadeIn.tsx
import {InView} from '@/components/motion-primitives/in-view'
import {ReactNode} from 'react'
import {Variants} from 'framer-motion'

interface FadeInProps {
  children: ReactNode
  delay?: number
  y?: number
  duration?: number
  className?: string
  // framer-motion overrides (D-07)
  initial?: Variants[string]
  animate?: Variants[string]
  exit?: Variants[string]
}

export function FadeIn({children, delay = 0, y = 8, duration = 0.35, className, ...rest}: FadeInProps) {
  return (
    <InView
      variants={{
        hidden: {opacity: 0, y, ...rest.initial},
        visible: {opacity: 1, y: 0, transition: {duration, delay, ease: [0.16, 1, 0.3, 1]}, ...rest.animate},
      }}
      className={className}
    >
      {children}
    </InView>
  )
}
```

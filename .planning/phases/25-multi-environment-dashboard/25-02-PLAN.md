       ```typescript
       const STORAGE_KEY = 'livos:docker:dashboard:selected-tag'

       export function deriveAllTags(envs: Pick&lt;Environment, 'tags'&gt;[]): string[] {
         const set = new Set&lt;string&gt;()
         for (const e of envs) for (const t of e.tags) set.add(t)
         return [...set].sort()
       }

       export function filterEnvs&lt;T extends {tags: string[]}&gt;(envs: T[], selected: string | null): T[] {
         if (!selected) return envs
         return envs.filter(e =&gt; e.tags.includes(selected))
       }

       export function useTagFilter() {
         const [selected, _setSelected] = useState&lt;string | null&gt;(() =&gt; {
           if (typeof window === 'undefined') return null
           return localStorage.getItem(STORAGE_KEY)
         })
         const setSelected = useCallback((tag: string | null) =&gt; {
           _setSelected(tag)
           if (tag) localStorage.setItem(STORAGE_KEY, tag)
           else localStorage.removeItem(STORAGE_KEY)
         }, [])
         return {selected, setSelected, deriveAllTags, filterEnvs}
       }
       ```

       ```typescript
       export interface TopCpuEntry {
         envId: string
         envName: string
         containerId: string
         containerName: string
         image: string
         cpuPercent: number
         memoryPercent: number
         isProtected: boolean
       }

       export const TOP_CPU_LIMIT = 10

       /**
        * Sort entries by cpuPercent desc; ties broken by envName asc, then containerName asc.
        * Caller is responsible for pre-filtering to running containers — this function
        * sorts whatever is given.
        */
       export function sortTopCpu(entries: TopCpuEntry[], limit: number = TOP_CPU_LIMIT): TopCpuEntry[] {
         return [...entries]
           .sort((a, b) =&gt; {
             if (b.cpuPercent !== a.cpuPercent) return b.cpuPercent - a.cpuPercent
             if (a.envName !== b.envName) return a.envName.localeCompare(b.envName)
             return a.containerName.localeCompare(b.containerName)
           })
           .slice(0, limit)
       }
       ```

       ```typescript
        * Bounded fanout: stats calls = envCount × 5 per 5s tick. With 5 envs
        * that's 25 stats calls/5s = 5/sec. True global Top-10-by-CPU would
        * require fanning out to ALL running containers, which scales linearly
        * with cluster size. v29 polish task may add a docker.allEnvCpuTop tRPC
        * aggregator (single round-trip per tick) when env+container counts grow.
        */
       const PER_ENV_CANDIDATES = 5
       const TOP_CPU_LIMIT = 10
       const POLL_MS = 5000

       export function useTopCpu(): {
         entries: TopCpuEntry[]
         isLoading: boolean
       } {
         const {data: envs} = useEnvironments()
         // For each env, fetch its running container list. Use trpcReact directly
         // — useEnvCardData would re-fetch images/volumes/etc which we don't need.
         const containerQueries = (envs ?? []).map(env =&gt; ({
           env,
           q: trpcReact.docker.listContainers.useQuery(
             {environmentId: env.id},
             {refetchInterval: POLL_MS, retry: false, staleTime: 2500},
           ),
         }))

         // Build candidate list: for each env, top-5 running by created desc.
         const candidates: Array&lt;{env: Environment; container: ContainerInfo}&gt; = []
         for (const {env, q} of containerQueries) {
           const running = (q.data ?? []).filter(c =&gt; c.state === 'running')
           const top5 = [...running].sort((a, b) =&gt; b.created - a.created).slice(0, PER_ENV_CANDIDATES)
           for (const c of top5) candidates.push({env, container: c})
         }

         // Fan out stats for each candidate. queryKey is [{name, environmentId}]
         // so React Query caches per-candidate.
         const statsQueries = candidates.map(({env, container}) =&gt; ({
           env,
           container,
           q: trpcReact.docker.containerStats.useQuery(
             {name: container.name, environmentId: env.id},
             {refetchInterval: POLL_MS, retry: false, staleTime: 2500},
           ),
         }))

         const entries: TopCpuEntry[] = statsQueries
           .filter(s =&gt; s.q.data !== undefined)
           .map(({env, container, q}) =&gt; ({
             envId: env.id,
             envName: env.name,
             containerId: container.id,
             containerName: container.name,
             image: container.image,
             cpuPercent: q.data!.cpuPercent,
             memoryPercent: q.data!.memoryPercent,
             isProtected: container.isProtected,
           }))

         return {
           entries: sortTopCpu(entries, TOP_CPU_LIMIT),
           isLoading: containerQueries.some(c =&gt; c.q.isLoading),
         }
       }
       ```
       NOTE: hooks-in-loops are normally a React rules violation, BUT each `containerQueries.map` iteration produces ONE useQuery call and the map iterates over a stable, hook-safe-on-mount source (`envs` from useEnvironments). React's hooks rule only forbids hooks inside CONDITIONAL branches; iterating a stable array is allowed when the array length is stable per render after envs hydrates. To make this airtight, gate the fanout on `envs !== undefined` — early-return `{entries: [], isLoading: true}` when envs is loading. This way the hook count is 0 until envs hydrates, then stable thereafter (envs are added/removed via discrete admin actions that trigger a full re-render). Document this in the hook header. If react detects hook-count drift between renders, the useEffect-based alternative is `useQueries` from React Query — fallback only if the executor sees the hooks-rule warning at runtime.

Replacement (THIS PLAN):

```typescript
// Phase 58 (FR-BROKER-C1-01): true token streaming — raw async iterator forwarding
res.setHeader('Content-Type', 'text/event-stream')
res.setHeader('Cache-Control', 'no-cache, no-transform')
res.setHeader('Connection', 'keep-alive')
res.setHeader('X-Accel-Buffering', 'no')
res.flushHeaders()

let stream
try {
  stream = await client.messages.create({...upstreamBody, stream: true} as any)
} catch (err) {
  // 401-on-create-stream → tryRefreshAndRetry preserved from Phase 57
  const refreshed = await tryRefreshAndRetry(err, token, makeClientFn, async (c) =>
    c.messages.create({...upstreamBody, stream: true} as any),
  )
  if (refreshed === null) throw mapApiError(err)
  stream = refreshed
}

for await (const event of stream) {
  if (res.writableEnded) break
  res.write(`event: ${(event as {type: string}).type}\n`)
  res.write(`data: ${JSON.stringify(event)}\n\n`)
  const flushable = res as unknown as {flush?: () => void}
  if (typeof flushable.flush === 'function') flushable.flush()
}
if (!res.writableEnded) res.end()
```

    ```bash
    #!/bin/bash
    set -euo pipefail

    # Phase 31 patch for /opt/livos/update.sh
    # Delivers:
    #   BUILD-01: verify_build() helper invoked after every pnpm/npm build
    #   BUILD-02: dist-copy loop over ALL @nexus+core* pnpm-store dirs
    #   BUILD-03: <root-cause-specific fix from 31-ROOT-CAUSE.md, if any>
    #
    # Idempotent: re-runs detect markers and exit 0 with ALREADY-PATCHED.
    # Apply via:  ssh <host> 'sudo bash -s' < phase31-update-sh-patch.sh

    UPDATE_SH="/opt/livos/update.sh"
    MARKER_VERIFY="# ── Phase 31 BUILD-01: verify_build helper ──"
    MARKER_LOOP="# ── Phase 31 BUILD-02: multi-dir dist-copy loop ──"
    MARKER_RC="# ── Phase 31 BUILD-03: root-cause fix ──"

    if [[ ! -f "$UPDATE_SH" ]]; then
        echo "ERROR: $UPDATE_SH not found on this host" >&2
        exit 1
    fi

    # Snapshot original (Phase 30 precedent — keep .pre-phaseNN backup)
    if [[ ! -f "$UPDATE_SH.pre-phase31" ]]; then
        cp "$UPDATE_SH" "$UPDATE_SH.pre-phase31"
        echo "Backup written: $UPDATE_SH.pre-phase31"
    fi

    # ─────────────────────────────────────────────────────────
    # Patch 1/3: BUILD-01 — insert verify_build() helper near top of script
    # ─────────────────────────────────────────────────────────
    if grep -q "$MARKER_VERIFY" "$UPDATE_SH"; then
        echo "ALREADY-PATCHED (BUILD-01): verify_build helper present"
    else
        # Anchor: the line right after the existing helper functions (step/ok/warn/fail)
        # Phase 30 confirmed these exist; splice immediately after the last `fail()` definition.
        ANCHOR_LINE=$(grep -n '^fail()' "$UPDATE_SH" | tail -1 | cut -d: -f1)
        if [[ -z "$ANCHOR_LINE" ]]; then
            echo "ERROR: cannot find fail() helper to anchor verify_build insertion" >&2
            exit 1
        fi
        # Find end of fail() function — first line matching `^}` after ANCHOR_LINE
        END_LINE=$(awk -v start="$ANCHOR_LINE" 'NR > start && /^}/ {print NR; exit}' "$UPDATE_SH")
        if [[ -z "$END_LINE" ]]; then
            echo "ERROR: cannot find end of fail() function" >&2
            exit 1
        fi
        INSERT_AT=$((END_LINE + 1))

        PATCH_BLOCK=$(mktemp)
        cat <<'EOPATCH' > "$PATCH_BLOCK"

    # ── Phase 31 BUILD-01: verify_build helper ──
    # Asserts that a build produced non-empty output. Call AFTER every build invocation.
    # Usage: verify_build "@livos/config" "/opt/livos/packages/config/dist"
    verify_build() {
        local pkg="$1"
        local outdir="$2"
        if [[ ! -d "$outdir" ]] || [[ -z "$(find "$outdir" -type f 2>/dev/null | head -1)" ]]; then
            echo "BUILD-FAIL: $pkg produced empty $outdir" >&2
            exit 1
        fi
        echo "[VERIFY] $pkg dist OK ($outdir)"
    }

    EOPATCH

        awk -v line="$INSERT_AT" -v patchfile="$PATCH_BLOCK" '
            BEGIN { while ((getline pl < patchfile) > 0) patch = patch pl ORS }
            NR == line { printf "%s", patch }
            { print }
        ' "$UPDATE_SH" > "$UPDATE_SH.new"
        mv "$UPDATE_SH.new" "$UPDATE_SH"
        chmod +x "$UPDATE_SH"
        rm -f "$PATCH_BLOCK"
        echo "PATCH-OK (BUILD-01): verify_build helper inserted at line $INSERT_AT"
    fi

    # ─────────────────────────────────────────────────────────
    # Patch 1b/3: insert verify_build CALLS after each build invocation
    # ─────────────────────────────────────────────────────────
    # Map of build invocation → expected output dir → package label
    # Each entry: "<grep-pattern-of-build-line>|<outdir>|<pkg-label>"
    BUILD_VERIFICATIONS=(
        "pnpm --filter @livos/config build|/opt/livos/packages/config/dist|@livos/config"
        "pnpm --filter ui build|/opt/livos/packages/ui/dist|@livos/ui"
        "npm run build --workspace=packages/core|/opt/nexus/packages/core/dist|@nexus/core"
        "npm run build --workspace=packages/worker|/opt/nexus/packages/worker/dist|@nexus/worker"
        "npm run build --workspace=packages/mcp-server|/opt/nexus/packages/mcp-server/dist|@nexus/mcp-server"
    )

    for entry in "${BUILD_VERIFICATIONS[@]}"; do
        IFS='|' read -r build_pat outdir pkg <<< "$entry"
        VERIFY_CALL="verify_build \"$pkg\" \"$outdir\""

        if grep -qF "$VERIFY_CALL" "$UPDATE_SH"; then
            echo "ALREADY-PATCHED (BUILD-01 call): $pkg verify already wired"
            continue
        fi

        # Find the build invocation line (last occurrence — accounts for any earlier reference in comments)
        BUILD_LINE=$(grep -nF "$build_pat" "$UPDATE_SH" | grep -v '^[[:space:]]*#' | tail -1 | cut -d: -f1)
        if [[ -z "$BUILD_LINE" ]]; then
            echo "WARN: build invocation '$build_pat' not found — skipping verify wiring for $pkg" >&2
            continue
        fi

        # Insert the verify call on the line AFTER the build invocation
        INSERT_AT=$((BUILD_LINE + 1))
        sed -i "${INSERT_AT}i\\${VERIFY_CALL}" "$UPDATE_SH"
        echo "PATCH-OK (BUILD-01 call): wired verify_build for $pkg after line $BUILD_LINE"
    done

    # ─────────────────────────────────────────────────────────
    # Patch 2/3: BUILD-02 — replace single-target dist-copy with loop
    # ─────────────────────────────────────────────────────────
    if grep -q "$MARKER_LOOP" "$UPDATE_SH"; then
        echo "ALREADY-PATCHED (BUILD-02): multi-dir dist-copy loop present"
    else
        # Anchor: the existing single-target find line.
        # The patch wraps the existing copy block with a for-loop over all matching dirs.
        OLD_FIND_LINE=$(grep -n "@nexus+core\*" "$UPDATE_SH" | grep -F "head -1" | head -1 | cut -d: -f1)
        if [[ -z "$OLD_FIND_LINE" ]]; then
            # Fallback: any line referencing @nexus+core* — Plan 01 snapshot will reveal the exact form
            OLD_FIND_LINE=$(grep -n "@nexus+core\*" "$UPDATE_SH" | head -1 | cut -d: -f1)
        fi
        if [[ -z "$OLD_FIND_LINE" ]]; then
            echo "ERROR: cannot find @nexus+core* dist-copy anchor in $UPDATE_SH" >&2
            exit 1
        fi

        # Find the end of the existing dist-copy block — heuristic: next blank line or next `step`/`ok` call
        END_LINE=$(awk -v start="$OLD_FIND_LINE" 'NR > start && (/^[[:space:]]*$/ || /^step / || /^ok /) {print NR; exit}' "$UPDATE_SH")
        if [[ -z "$END_LINE" ]]; then
            END_LINE=$((OLD_FIND_LINE + 5))
        fi

        # Replace lines [OLD_FIND_LINE .. END_LINE-1] with the new loop block
        REPLACEMENT=$(mktemp)
        cat <<'EOLOOP' > "$REPLACEMENT"
    # ── Phase 31 BUILD-02: multi-dir dist-copy loop ──
    NEXUS_CORE_DIST_SRC="/opt/nexus/packages/core/dist"
    if [[ ! -d "$NEXUS_CORE_DIST_SRC" ]] || [[ -z "$(find "$NEXUS_CORE_DIST_SRC" -type f | head -1)" ]]; then
        echo "DIST-COPY-FAIL: source $NEXUS_CORE_DIST_SRC is empty — nexus core build did not emit" >&2
        exit 1
    fi
    COPY_COUNT=0
    for store_dir in /opt/livos/node_modules/.pnpm/@nexus+core*/; do
        [[ -d "$store_dir" ]] || continue
        target="${store_dir}node_modules/@nexus/core/dist"
        mkdir -p "$(dirname "$target")"
        rm -rf "$target"
        cp -r "$NEXUS_CORE_DIST_SRC" "$target"
        if [[ -z "$(find "$target" -type f 2>/dev/null | head -1)" ]]; then
            echo "DIST-COPY-FAIL: post-copy target $target is empty" >&2
            exit 1
        fi
        COPY_COUNT=$((COPY_COUNT + 1))
        echo "[VERIFY] nexus core dist copied to $store_dir"
    done
    if [[ "$COPY_COUNT" -eq 0 ]]; then
        echo "DIST-COPY-FAIL: no @nexus+core* dirs found under /opt/livos/node_modules/.pnpm/" >&2
        exit 1
    fi
    echo "[VERIFY] nexus core dist copied to $COPY_COUNT pnpm-store dir(s)"
    EOLOOP

        # Splice: keep lines 1..OLD_FIND_LINE-1, insert REPLACEMENT, keep lines END_LINE..end
        head -n $((OLD_FIND_LINE - 1)) "$UPDATE_SH" > "$UPDATE_SH.new"
        cat "$REPLACEMENT" >> "$UPDATE_SH.new"
        tail -n +"$END_LINE" "$UPDATE_SH" >> "$UPDATE_SH.new"
        mv "$UPDATE_SH.new" "$UPDATE_SH"
        chmod +x "$UPDATE_SH"
        rm -f "$REPLACEMENT"
        echo "PATCH-OK (BUILD-02): dist-copy loop spliced (replaced lines $OLD_FIND_LINE..$((END_LINE - 1)))"
    fi

    # ─────────────────────────────────────────────────────────
    # Patch 3/3: BUILD-03 — root-cause-specific fix
    # ─────────────────────────────────────────────────────────
    # READ 31-ROOT-CAUSE.md "Recommended Remediation For Plan 02" section.
    # If verdict is "inconclusive" → this section can be a no-op block:
    if grep -q "$MARKER_RC" "$UPDATE_SH"; then
        echo "ALREADY-PATCHED (BUILD-03): root-cause fix marker present"
    else
        # Example for "set -euo pipefail missing" verdict:
        if ! head -5 "$UPDATE_SH" | grep -q 'set -euo pipefail'; then
            sed -i '2i set -euo pipefail\n# '"$MARKER_RC"' set -euo pipefail enforced (BUILD-03)' "$UPDATE_SH"
            echo "PATCH-OK (BUILD-03): set -euo pipefail enforced at line 2"
        else
            # Inconclusive case — write only the marker comment so idempotency works
            sed -i "2i # ${MARKER_RC} (verdict: inconclusive — BUILD-01 fail-loud guard is the safety net)" "$UPDATE_SH"
            echo "PATCH-OK (BUILD-03): marker comment recorded (no script change needed)"
        fi
    fi

    # ─────────────────────────────────────────────────────────
    # Final syntax check on the patched script
    # ─────────────────────────────────────────────────────────
    if ! bash -n "$UPDATE_SH"; then
        echo "FATAL: patched $UPDATE_SH failed bash syntax check — restoring backup" >&2
        cp "$UPDATE_SH.pre-phase31" "$UPDATE_SH"
        exit 1
    fi

    echo
    echo "=== PHASE 31 PATCH COMPLETE ==="
    echo "Backup:  $UPDATE_SH.pre-phase31"
    echo "Patched: $UPDATE_SH"
    echo "Markers present:"
    grep -n -E "Phase 31 BUILD-(01|02|03)" "$UPDATE_SH" | head -10 || true
    ```

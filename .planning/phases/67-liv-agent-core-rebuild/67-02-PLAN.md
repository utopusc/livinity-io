export class LivAgentRunner {
  private snapshots = new Map<string, ToolCallSnapshot>();
  private currentRunId: string | null = null;
  private stopRequested = false;

  constructor(private readonly opts: LivAgentRunnerOptions) {}

  async start(runId: string, task: string): Promise<void> {
    this.currentRunId = runId;
    this.stopRequested = false;
    await this.opts.runStore.appendChunk(runId, { type: 'status', payload: 'running' });

    // Wire SDK event listeners (read sdk-agent-runner.ts ONCE for event names)
    const onEvent = async (event: SdkEvent) => { ... };
    this.opts.sdkRunner.on(/*event-name*/, onEvent);

    try {
      // Cooperative stop: poll between iters OR check on every event handler entry
      // Implementation choice (Claude's discretion): wrap sdkRunner.run(task) in a Promise,
      // and on each event handler call check getControl(runId). If 'stop' → call sdkRunner.stop?.()
      // (if exposed) AND remove our listener AND resolve the start() promise.
      await this.opts.sdkRunner.run(task);
      // ...completion path...
      const finalResult = /* extract from final SDK message or {} */;
      await this.opts.runStore.markComplete(runId, finalResult);
    } catch (err) {
      const error = err as Error;
      await this.opts.runStore.appendChunk(runId, {
        type: 'error',
        payload: { message: error.message }
      });
      await this.opts.runStore.markError(runId, {
        message: error.message,
        stack: error.stack
      });
    } finally {
      this.opts.sdkRunner.off?.(/*event-name*/, onEvent);
      this.currentRunId = null;
    }
  }

  async stop(runId: string): Promise<void> {
    await this.opts.runStore.setControl(runId, 'stop');
    this.stopRequested = true;
  }

  // Internal helpers (private):
  private async handleAssistantMessage(runId: string, msg: AssistantMessage): Promise<void> {
    // Extract reasoning_content (Kimi-style) — emit reasoning chunk FIRST
    if (msg.reasoning_content) {
      await this.opts.runStore.appendChunk(runId, {
        type: 'reasoning',
        payload: msg.reasoning_content,
      });
    }
    // Extract text blocks — emit as text chunks
    for (const block of msg.content ?? []) {
      if (block.type === 'text' && block.text) {
        await this.opts.runStore.appendChunk(runId, {
          type: 'text',
          payload: block.text,
        });
      }
      if (block.type === 'tool_use') {
        await this.openToolSnapshot(runId, block);
      }
    }
  }

  private async openToolSnapshot(runId: string, block: ToolUseBlock): Promise<void> {
    const category = categorizeTool(block.name);
    const snapshot: ToolCallSnapshot = {
      toolId: block.id,
      toolName: block.name,
      category,
      assistantCall: { input: block.input, ts: Date.now() },
      status: 'running',
      startedAt: Date.now(),
    };
    this.snapshots.set(block.id, snapshot);
    await this.opts.runStore.appendChunk(runId, { type: 'tool_snapshot', payload: snapshot });

    // D-16 stub: short-circuit computer-use tools
    if (category === 'computer-use') {
      console.warn(`[LivAgentRunner] computer-use tool ${block.name} stubbed until P71`);
      const stubbed: ToolCallSnapshot = {
        ...snapshot,
        toolResult: {
          output: 'Computer use not available until P71 (Bytebot integration)',
          isError: true,
          ts: Date.now(),
        },
        status: 'error',
        completedAt: Date.now(),
      };
      this.snapshots.set(block.id, stubbed);
      await this.opts.runStore.appendChunk(runId, { type: 'tool_snapshot', payload: stubbed });
    }
  }

  private async closeToolSnapshot(runId: string, result: ToolResultBlock): Promise<void> {
    const existing = this.snapshots.get(result.tool_use_id);
    if (!existing) return; // orphan tool_result — log and skip
    const merged: ToolCallSnapshot = {
      ...existing,
      toolResult: { output: result.content, isError: result.is_error ?? false, ts: Date.now() },
      status: result.is_error ? 'error' : 'done',
      completedAt: Date.now(),
    };
    this.snapshots.set(result.tool_use_id, merged);
    await this.opts.runStore.appendChunk(runId, { type: 'tool_snapshot', payload: merged });
  }
}

export function categorizeTool(toolName: string): ToolCategory {
  if (toolName.startsWith('computer_use_') || toolName.startsWith('bytebot_')) return 'computer-use';
  if (toolName.startsWith('mcp_') || toolName.startsWith('mcp-')) return 'mcp';
  if (toolName.startsWith('browser-') || toolName.startsWith('browser_')) return 'browser';
  if (toolName === 'execute-command' || toolName.startsWith('terminal-') || toolName.startsWith('terminal_')) return 'terminal';
  if (toolName === 'str-replace' || toolName === 'edit-file' || toolName.startsWith('str_replace')) return 'fileEdit';
  if (toolName === 'read-file' || toolName === 'write-file' || toolName === 'list-files' || toolName === 'delete-file' ||
      toolName.startsWith('read_') || toolName.startsWith('write_') || toolName.startsWith('list_files') || toolName.startsWith('delete_')) return 'file';
  if (toolName === 'web-search' || toolName.startsWith('web_search')) return 'webSearch';
  if (toolName === 'web-crawl' || toolName.startsWith('web_crawl')) return 'webCrawl';
  if (toolName === 'web-scrape' || toolName.startsWith('web_scrape')) return 'webScrape';
  return 'generic';
}

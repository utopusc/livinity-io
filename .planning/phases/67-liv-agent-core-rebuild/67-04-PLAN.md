**applyChunk (Zustand store mutation — the heart of the hook):**

```typescript
applyChunk: (convId, chunk) => set(state => {
  const conv = state.conversations.get(convId);
  if (!conv) return state;
  const next = { ...conv, lastSeenIdx: Math.max(conv.lastSeenIdx, chunk.idx) };

  if (chunk.type === 'text') {
    const last = next.messages[next.messages.length - 1];
    if (last && last.role === 'assistant') {
      next.messages = [...next.messages.slice(0, -1), { ...last, text: last.text + (chunk.payload as string) }];
    } else {
      next.messages = [...next.messages, { id: `msg_${conv.runId}_${chunk.idx}`, role: 'assistant', text: chunk.payload as string, ts: chunk.ts }];
    }
  } else if (chunk.type === 'reasoning') {
    const last = next.messages[next.messages.length - 1];
    if (last && last.role === 'assistant') {
      next.messages = [...next.messages.slice(0, -1), { ...last, reasoning: (last.reasoning ?? '') + (chunk.payload as string) }];
    } else {
      next.messages = [...next.messages, { id: `msg_${conv.runId}_${chunk.idx}`, role: 'assistant', text: '', reasoning: chunk.payload as string, ts: chunk.ts }];
    }
  } else if (chunk.type === 'tool_snapshot') {
    const snap = chunk.payload as ToolCallSnapshot;
    const newSnaps = new Map(next.snapshots);
    newSnaps.set(snap.toolId, snap);  // replaces — D-15 dedupe semantics
    next.snapshots = newSnaps;
  } else if (chunk.type === 'error') {
    next.status = 'error';
    next.errorMessage = (chunk.payload as { message: string }).message;
  } else if (chunk.type === 'status') {
    const s = chunk.payload as string;
    if (s === 'complete' || s === 'error' || s === 'stopped') next.status = s as StreamStatus;
  }

  const map = new Map(state.conversations);
  map.set(convId, next);
  return { conversations: map };
}),
```

**Public hook (final shape):**
```typescript
export function useLivAgentStream({ conversationId, autoStart }: UseLivAgentStreamOpts): UseLivAgentStreamReturn {
  // 1. Ensure conversation slice exists
  // 2. Subscribe via useLivAgentStore selector
  // 3. eventSourceRef + reconnectTimerRef refs
  // 4. openStream(runId, fromIdx) callback — opens new EventSource, wires onmessage / addEventListener('complete') / onerror
  // 5. sendMessage callback — POST /api/agent/start, store runId, openStream(runId, -1)
  // 6. stop callback — POST /api/agent/runs/${runId}/control with { signal: 'stop' }
  // 7. retry callback — clear timer, reset reconnect counter, openStream(runId, lastSeenIdx)
  // 8. autoStart effect — re-open EventSource if runId exists for this conv on mount
  // 9. Cleanup effect — close EventSource, clear timer on unmount
  return { messages, snapshots, status, sendMessage, stop, runId, retry };
}
```

```typescript
import express, { type Request, type Response } from 'express';
import { Redis } from 'ioredis';
import { RunStore, LivAgentRunner, type Chunk, type RunMeta } from '@nexus/core';
import { isAuthenticated } from '../auth/is-authenticated.js'; // OR whatever the existing middleware path is
import { /* SdkAgentRunner factory from existing daemon DI */ } from '../../...';

// Module-level singletons (one per daemon process)
let runStore: RunStore | null = null;
let livAgentRunner: LivAgentRunner | null = null;

export function registerAgentRuns(app: express.Application, redis: Redis, sdkRunnerFactory: () => SdkAgentRunner, toolRegistry: ToolRegistry) {
  if (!runStore) runStore = new RunStore(redis);
  if (!livAgentRunner) livAgentRunner = new LivAgentRunner({
    runStore,
    sdkRunner: sdkRunnerFactory(),  // OR per-run if SDK runners are not reusable — Claude's discretion
    toolRegistry,
    redisClient: redis,
  });

  // POST /api/agent/start
  app.post('/api/agent/start', isAuthenticated, async (req: Request, res: Response) => {
    const userId = req.user?.userId; // assuming middleware attaches req.user
    if (!userId) return res.status(401).json({ error: 'unauthenticated' });
    const { task, conversationId } = req.body ?? {};
    if (!task || typeof task !== 'string' || task.trim().length === 0) {
      return res.status(400).json({ error: 'task is required' });
    }
    const runId = await runStore!.createRun(userId, task);
    // Fire-and-forget runner — D-18 no BullMQ
    livAgentRunner!.start(runId, task).catch(err => {
      console.error(`[agent-runs] runner failed for ${runId}:`, err);
    });
    res.status(200).json({ runId, sseUrl: `/api/agent/runs/${runId}/stream` });
  });

  // GET /api/agent/runs/:runId/stream
  app.get('/api/agent/runs/:runId/stream', async (req: Request, res: Response) => {
    // Auth: header OR ?token= query
    const headerToken = req.headers.authorization?.replace(/^Bearer\s+/i, '');
    const queryToken = typeof req.query.token === 'string' ? req.query.token : undefined;
    const token = headerToken ?? queryToken;
    if (!token) return res.status(401).json({ error: 'unauthenticated' });
    let userId: string;
    try { userId = verifyJwt(token).userId; } catch { return res.status(401).json({ error: 'invalid token' }); }

    const { runId } = req.params;
    const meta = await runStore!.getMeta(runId);
    if (!meta) return res.status(404).json({ error: 'run not found' });
    if (meta.userId !== userId) return res.status(403).json({ error: 'forbidden' });

    const after = parseInt((req.query.after as string) ?? '-1', 10);  // default -1 = "send everything"
    const fromIndex = isNaN(after) ? 0 : after + 1;

    res.writeHead(200, {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache, no-transform',
      'Connection': 'keep-alive',
      'X-Accel-Buffering': 'no',
    });

    // 1. Catch-up
    const catchup = await runStore!.getChunks(runId, fromIndex);
    for (const chunk of catchup) {
      res.write(`data: ${JSON.stringify(chunk)}\n\n`);
    }

    // 2. If terminal, send complete event and end
    const finalMeta = await runStore!.getMeta(runId);
    if (finalMeta && (finalMeta.status === 'complete' || finalMeta.status === 'error' || finalMeta.status === 'stopped')) {
      res.write(`event: complete\ndata: ${JSON.stringify({ status: finalMeta.status })}\n\n`);
      res.end();
      return;
    }

    // 3. Subscribe for live tail
    const unsubscribe = await runStore!.subscribeChunks(runId, (chunk) => {
      res.write(`data: ${JSON.stringify(chunk)}\n\n`);
      if (chunk.type === 'status' && (chunk.payload === 'complete' || chunk.payload === 'error' || chunk.payload === 'stopped')) {
        // graceful close after terminal status chunk
        res.write(`event: complete\ndata: ${JSON.stringify({ status: chunk.payload })}\n\n`);
        res.end();
      }
    });

    // 4. Heartbeat every 15s
    const heartbeat = setInterval(() => {
      res.write(`: heartbeat\n\n`);
    }, 15000);

    // 5. Cleanup on disconnect
    const cleanup = async () => {
      clearInterval(heartbeat);
      await unsubscribe();
    };
    req.on('close', () => { cleanup().catch(() => {}); });
    req.on('aborted', () => { cleanup().catch(() => {}); });
  });

  // POST /api/agent/runs/:runId/control
  app.post('/api/agent/runs/:runId/control', isAuthenticated, async (req: Request, res: Response) => {
    const userId = req.user?.userId;
    if (!userId) return res.status(401).json({ error: 'unauthenticated' });
    const { runId } = req.params;
    const meta = await runStore!.getMeta(runId);
    if (!meta) return res.status(404).json({ error: 'run not found' });
    if (meta.userId !== userId) return res.status(403).json({ error: 'forbidden' });
    const signal = req.body?.signal;
    if (signal !== 'stop') return res.status(400).json({ error: 'invalid signal' });
    await runStore!.setControl(runId, 'stop');
    res.status(200).json({ ok: true });
  });
}
```

| livinityd → @nexus/core | Trusted (same process); no marshalling concerns. |

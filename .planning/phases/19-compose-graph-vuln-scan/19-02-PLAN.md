    Structure:
    ```typescript
    import Dockerode from 'dockerode'
    import {Redis} from 'ioredis'
    import type {VulnScanResult, CveEntry, Severity} from './types.js'

    const docker = new Dockerode()
    const TRIVY_IMAGE = 'aquasec/trivy:latest'
    const CACHE_PREFIX = 'nexus:vuln:'
    const CACHE_TTL_SECONDS = 60 * 60 * 24 * 7   // 7 days
    const SEVERITY_ORDER: Severity[] = ['CRITICAL', 'HIGH', 'MEDIUM', 'LOW']

    let _redis: Redis | null = null
    function getRedis(): Redis {
      if (!_redis) _redis = new Redis(process.env.REDIS_URL || 'redis://localhost:6379', {maxRetriesPerRequest: null})
      return _redis
    }

    async function resolveDigest(imageRef: string): Promise<string> { ... }   // returns "sha256:..."
    async function ensureTrivyImage(): Promise<void> { ... }                   // pulls aquasec/trivy if absent
    async function runTrivyAndCollectStdout(imageRef: string): Promise<string> { ... }
    function parseTrivyJson(raw: string, imageRef: string, digest: string, trivyVersion: string): VulnScanResult { ... }

    export async function getCachedScan(imageRef: string): Promise<VulnScanResult | null> { ... }
    export async function scanImage(imageRef: string, force?: boolean): Promise<VulnScanResult> { ... }
    ```

    Implementation details:

    - `resolveDigest(imageRef)`: `const info = await docker.getImage(imageRef).inspect(); return info.Id` — `info.Id` is `sha256:<hex>`. Errors: 404 → throw `[image-not-found] Image not found: ${imageRef}`.

    - `ensureTrivyImage()`: try `docker.getImage(TRIVY_IMAGE).inspect()`; on 404, await `docker.pull(TRIVY_IMAGE, ...)` using `docker.modem.followProgress` — mirror `pullImage` in docker.ts lines 528–561.

    - `runTrivyAndCollectStdout(imageRef)`: use `dockerode.run` to capture both stdout and stderr. Two valid implementations — pick ONE (recommend execa for simpler stdout capture):
      - **Preferred (execa)**: `import {$} from 'execa'`. Run `await $({reject: false})\`docker run --rm -v /var/run/docker.sock:/var/run/docker.sock ${TRIVY_IMAGE} image ${imageRef} --format json --severity CRITICAL,HIGH,MEDIUM,LOW --quiet\``. Trivy returns exit code 0 even when CVEs are found (success). Non-zero → throw `[trivy-failed] ${result.stderr}`. Return `result.stdout`.
      - Capture timeout: 5 minutes (`{timeout: 300_000}`). On timeout → throw `[trivy-timeout] Scan exceeded 5 minutes`.

    - `parseTrivyJson(raw, imageRef, digest, trivyVersion)`:
      - `const json = JSON.parse(raw)` — wrap in try/catch, on failure throw `[trivy-parse] Could not parse Trivy output (first 200 chars: ${raw.slice(0,200)})`.
      - Trivy JSON shape: `{SchemaVersion, ArtifactName, Results: [{Target, Type, Vulnerabilities: [...]}]}`. Vulnerabilities can be missing/empty per Result.
      - For each `Result.Vulnerabilities` entry, build a CveEntry:
        - id = `vuln.VulnerabilityID`
        - severity = `vuln.Severity` (already CRITICAL/HIGH/MEDIUM/LOW); skip entries with severity not in SEVERITY_ORDER (UNKNOWN gets dropped per CGV-02 — only the four documented levels are surfaced)
        - packageName = `vuln.PkgName ?? ''`
        - installedVersion = `vuln.InstalledVersion ?? ''`
        - fixedVersion = `vuln.FixedVersion || null`
        - cvss = best of `vuln.CVSS?.nvd?.V3Score ?? vuln.CVSS?.redhat?.V3Score ?? vuln.CVSS?.ghsa?.V3Score ?? null`
        - title = `vuln.Title ?? vuln.VulnerabilityID`
        - description = `(vuln.Description ?? '').slice(0, 500)`
        - primaryUrl = `vuln.PrimaryURL ?? null`
      - Build `counts: {CRITICAL: 0, HIGH: 0, MEDIUM: 0, LOW: 0}` and increment per CVE.
      - Sort CVEs: severity rank (SEVERITY_ORDER index) ASC, then cvss DESC, then id ASC for stability.
      - Return `{imageRef, imageDigest: digest, scannedAt: Date.now(), counts, cves, trivyVersion, cached: false}`.

    - `getCachedScan(imageRef)`:
      - `const digest = await resolveDigest(imageRef)`
      - `const raw = await getRedis().get(CACHE_PREFIX + digest.replace(/^sha256:/, ''))`
      - If null → return null. Else `const result = JSON.parse(raw); result.cached = true; return result`.

    - `scanImage(imageRef, force)`:
      - `const digest = await resolveDigest(imageRef)` — establishes the cache key BEFORE any expensive work.
      - If `!force`: check cache. Hit → return immediately with `cached: true`.
      - `await ensureTrivyImage()` (pulls trivy on first scan ever).
      - `const raw = await runTrivyAndCollectStdout(imageRef)` — note: pass the original `imageRef` (the tag), not the digest, because Trivy resolves it internally.
      - `const trivyVersion = JSON.parse(raw).SchemaVersion?.toString() ?? 'unknown'`
      - `const result = parseTrivyJson(raw, imageRef, digest, trivyVersion)`
      - `await getRedis().set(CACHE_PREFIX + digest.replace(/^sha256:/, ''), JSON.stringify({...result, cached: false}), 'EX', CACHE_TTL_SECONDS)` — store with `cached: false` so subsequent reads can flip the flag without mutating storage.
      - Return result.

    Error code conventions (must match docker.ts):
    `[image-not-found]`, `[trivy-failed]`, `[trivy-timeout]`, `[trivy-parse]`, `[trivy-unavailable]` (when ensureTrivyImage's pull fails).

    ```typescript
    scanImage: adminProcedure
      .input(z.object({
        imageRef: z.string().min(1).max(500),
        force: z.boolean().optional().default(false),
      }))
      .mutation(async ({input}) => {
        try {
          return await scanImage(input.imageRef, input.force)
        } catch (err: any) {
          if (err.message?.includes('[image-not-found]')) {
            throw new TRPCError({code: 'NOT_FOUND', message: err.message.replace('[image-not-found] ', '')})
          }
          if (err.message?.includes('[trivy-timeout]')) {
            throw new TRPCError({code: 'TIMEOUT', message: err.message.replace('[trivy-timeout] ', '')})
          }
          if (err.message?.includes('[trivy-failed]') || err.message?.includes('[trivy-parse]') || err.message?.includes('[trivy-unavailable]')) {
            throw new TRPCError({code: 'INTERNAL_SERVER_ERROR', message: err.message.replace(/^\[[^\]]+\] /, '')})
          }
          throw new TRPCError({code: 'INTERNAL_SERVER_ERROR', message: err.message || `Failed to scan image ${input.imageRef}`})
        }
      }),

    getCachedScan: adminProcedure
      .input(z.object({imageRef: z.string().min(1).max(500)}))
      .query(async ({input}) => {
        try {
          return await getCachedScan(input.imageRef)
        } catch (err: any) {
          if (err.message?.includes('[image-not-found]')) {
            throw new TRPCError({code: 'NOT_FOUND', message: err.message.replace('[image-not-found] ', '')})
          }
          throw new TRPCError({code: 'INTERNAL_SERVER_ERROR', message: err.message || `Failed to get cached scan for ${input.imageRef}`})
        }
      }),
    ```

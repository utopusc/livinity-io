```tsx
import {Link, useLocation} from 'react-router-dom'
import {
    RiHomeLine,
    RiHomeFill,
    RiAppsLine,
    RiAppsFill,
    RiChat3Line,
    RiChat3Fill,
    RiFolderLine,
    RiFolderFill,
    RiSettings4Line,
    RiSettings4Fill,
} from 'react-icons/ri'
import {cn} from '@/shadcn-lib/utils'
import {useIsMobile} from '@/hooks/use-is-mobile'

const tabs = [
    {label: 'Home', to: '/', icon: RiHomeLine, activeIcon: RiHomeFill, match: (p: string) => p === '/'},
    {label: 'Apps', to: '/app-store', icon: RiAppsLine, activeIcon: RiAppsFill, match: (p: string) => p.startsWith('/app-store')},
    {label: 'AI', to: '/ai-chat', icon: RiChat3Line, activeIcon: RiChat3Fill, match: (p: string) => p.startsWith('/ai-chat')},
    {label: 'Files', to: '/files/Home', icon: RiFolderLine, activeIcon: RiFolderFill, match: (p: string) => p.startsWith('/files')},
    {label: 'Settings', to: '/settings', icon: RiSettings4Line, activeIcon: RiSettings4Fill, match: (p: string) => p.startsWith('/settings')},
]

export function MobileTabBar() {
    const {pathname} = useLocation()
    const isMobile = useIsMobile()

    if (!isMobile) return null

    return (
        <nav className='fixed bottom-0 left-0 right-0 z-50 flex items-center border-t border-border-subtle bg-[#111111] px-2 pb-[env(safe-area-inset-bottom)]'>
            {tabs.map((tab) => {
                const isActive = tab.match(pathname)
                const Icon = isActive ? tab.activeIcon : tab.icon
                return (
                    <Link
                        key={tab.label}
                        to={tab.to}
                        className={cn(
                            'flex flex-1 flex-col items-center gap-0.5 py-2 transition-colors',
                            isActive ? 'text-brand' : 'text-text-tertiary',
                        )}
                    >
                        <Icon className='h-6 w-6' />
                        <span className='text-[10px] font-medium'>{tab.label}</span>
                    </Link>
                )
            })}
        </nav>
    )
}

export function MobileTabBarSpacer() {
    const isMobile = useIsMobile()
    if (!isMobile) return null
    return <div className='h-16' />
}
```

```typescript
import {describe, expect, test} from 'vitest'
import {readFileSync} from 'node:fs'
import {join, dirname} from 'node:path'
import {fileURLToPath} from 'node:url'

describe('Phase 44 schema migration — broker_usage idempotency (D-44-20)', () => {
  test('schema.sql contains broker_usage CREATE TABLE IF NOT EXISTS', () => {
    const schemaPath = join(dirname(fileURLToPath(import.meta.url)), '..', 'database', 'schema.sql')
    const sql = readFileSync(schemaPath, 'utf8')
    expect(sql).toMatch(/CREATE TABLE IF NOT EXISTS broker_usage/)
    expect(sql).toMatch(/CREATE INDEX IF NOT EXISTS idx_broker_usage_user_created/)
    expect(sql).toMatch(/CREATE INDEX IF NOT EXISTS idx_broker_usage_app/)
  })

  test('broker_usage table definition references users(id) ON DELETE CASCADE', () => {
    const schemaPath = join(dirname(fileURLToPath(import.meta.url)), '..', 'database', 'schema.sql')
    const sql = readFileSync(schemaPath, 'utf8')
    // Extract the broker_usage CREATE TABLE block
    const match = sql.match(/CREATE TABLE IF NOT EXISTS broker_usage[\s\S]*?\);/)
    expect(match).toBeTruthy()
    expect(match![0]).toMatch(/REFERENCES users\(id\) ON DELETE CASCADE/)
  })

  test('schema.sql is idempotent — running twice does not produce duplicate table / index errors', async () => {
    // If a real test pool helper exists in livinityd, prefer it. Otherwise this
    // becomes a string-only assertion (the IF NOT EXISTS guards are visible in
    // the SQL — that's the contract).
    const schemaPath = join(dirname(fileURLToPath(import.meta.url)), '..', 'database', 'schema.sql')
    const sql = readFileSync(schemaPath, 'utf8')
    // String-level idempotency check: every CREATE TABLE / CREATE INDEX in the
    // broker_usage block uses IF NOT EXISTS.
    const brokerBlock = sql.slice(sql.indexOf('-- Broker Usage'))
    const creates = brokerBlock.match(/CREATE (TABLE|INDEX)[^\n]*/g) ?? []
    expect(creates.length).toBeGreaterThanOrEqual(3) // 1 table + 2 indexes
    for (const c of creates) {
      expect(c).toMatch(/IF NOT EXISTS/)
    }
  })
})
```

**Step 1 — Create integration.test.ts:**

This test validates D-44-22: end-to-end through the capture middleware. Strategy: build a stub Express app, mount the real capture middleware + a stub broker handler that returns either an Anthropic-shaped sync response, an OpenAI sync response, an SSE stream, or a 429. Use supertest to drive the request and assert insertUsage was called with the expected row.

```typescript
import {afterEach, beforeEach, describe, expect, test, vi} from 'vitest'
import express from 'express'
import request from 'supertest'

import {createCaptureMiddleware} from './capture-middleware.js'
import {_clearContainerResolverCache} from './container-resolver.js'

// Mock the database layer — we assert insertUsage call shape, not real DB writes.
const insertUsageMock = vi.fn().mockResolvedValue(undefined)
vi.mock('./database.js', () => ({
  insertUsage: (...args: unknown[]) => insertUsageMock(...args),
}))

const fakeLivinityd = {
  logger: {log: vi.fn(), warn: vi.fn(), error: vi.fn(), info: vi.fn()},
} as unknown as Parameters<typeof createCaptureMiddleware>[0]

function buildApp(brokerHandler: express.RequestHandler): express.Application {
  const app = express()
  app.use(express.json())
  app.use('/u/:userId/v1', createCaptureMiddleware(fakeLivinityd))
  app.post('/u/:userId/v1/messages', brokerHandler)
  app.post('/u/:userId/v1/chat/completions', brokerHandler)
  return app
}

describe('Phase 44 integration — capture middleware + broker handler -> insertUsage', () => {
  beforeEach(() => {
    insertUsageMock.mockClear()
    _clearContainerResolverCache()
  })

  test('Anthropic sync response triggers insertUsage with prompt_tokens + completion_tokens (FR-DASH-01)', async () => {
    const app = buildApp((req, res) => {
      res.json({
        id: 'msg_abc',
        type: 'message',
        role: 'assistant',
        content: [{type: 'text', text: 'hi'}],
        model: 'claude-sonnet-4-6',
        stop_reason: 'end_turn',
        stop_sequence: null,
        usage: {input_tokens: 42, output_tokens: 13},
      })
    })

    const res = await request(app).post('/u/USERID/v1/messages').send({model: 'claude-sonnet-4-6', messages: []})
    expect(res.status).toBe(200)

    // Allow the fire-and-forget recordRow promise to resolve
    await new Promise((r) => setImmediate(r))

    expect(insertUsageMock).toHaveBeenCalledTimes(1)
    const call = insertUsageMock.mock.calls[0][0]
    expect(call.userId).toBe('USERID')
    expect(call.promptTokens).toBe(42)
    expect(call.completionTokens).toBe(13)
    expect(call.endpoint).toBe('messages')
    expect(call.requestId).toBe('msg_abc')
    expect(call.model).toBe('claude-sonnet-4-6')
  })

  test('OpenAI sync response triggers insertUsage with endpoint=chat-completions', async () => {
    const app = buildApp((req, res) => {
      res.json({
        id: 'chatcmpl-xyz',
        choices: [{message: {role: 'assistant', content: 'hi'}, finish_reason: 'stop'}],
        model: 'gpt-4',
        usage: {prompt_tokens: 7, completion_tokens: 5, total_tokens: 12},
      })
    })

    await request(app).post('/u/USERID/v1/chat/completions').send({model: 'gpt-4', messages: []})
    await new Promise((r) => setImmediate(r))

    expect(insertUsageMock).toHaveBeenCalledTimes(1)
    const call = insertUsageMock.mock.calls[0][0]
    expect(call.endpoint).toBe('chat-completions')
    expect(call.promptTokens).toBe(7)
    expect(call.completionTokens).toBe(5)
  })

  test('429 response triggers insertUsage with endpoint=429-throttled (FR-DASH-03)', async () => {
    const app = buildApp((req, res) => {
      res.status(429).set('Retry-After', '60').json({error: {type: 'rate_limit_error'}})
    })

    const res = await request(app).post('/u/USERID/v1/messages').send({model: 'claude-sonnet-4-6', messages: []})
    expect(res.status).toBe(429)
    expect(res.headers['retry-after']).toBe('60')

    await new Promise((r) => setImmediate(r))
    expect(insertUsageMock).toHaveBeenCalledTimes(1)
    const call = insertUsageMock.mock.calls[0][0]
    expect(call.endpoint).toBe('429-throttled')
    expect(call.promptTokens).toBe(0)
    expect(call.completionTokens).toBe(0)
  })

  test('Malformed response (no usage object) does NOT call insertUsage and does not crash', async () => {
    const app = buildApp((req, res) => {
      res.json({whatever: 'no usage here'})
    })

    const res = await request(app).post('/u/USERID/v1/messages').send({model: 'claude-sonnet-4-6', messages: []})
    expect(res.status).toBe(200)

    await new Promise((r) => setImmediate(r))
    expect(insertUsageMock).not.toHaveBeenCalled()
  })

  test('SSE Anthropic response — insertUsage called once at stream end with accumulated tokens', async () => {
    const app = buildApp((req, res) => {
      res.setHeader('Content-Type', 'text/event-stream')
      res.write('event: message_start\ndata: {"type":"message_start","message":{"id":"msg_sse","model":"claude-sonnet-4-6","usage":{"input_tokens":20}}}\n\n')
      res.write('event: content_block_delta\ndata: {"type":"content_block_delta","delta":{"text":"hello"}}\n\n')
      res.write('event: message_delta\ndata: {"type":"message_delta","delta":{"stop_reason":"end_turn"},"usage":{"output_tokens":8}}\n\n')
      res.end()
    })

    const res = await request(app).post('/u/USERID/v1/messages').send({model: 'claude-sonnet-4-6', stream: true, messages: []})
    expect(res.status).toBe(200)

    await new Promise((r) => setImmediate(r))
    expect(insertUsageMock).toHaveBeenCalledTimes(1)
    const call = insertUsageMock.mock.calls[0][0]
    expect(call.endpoint).toBe('messages')
    expect(call.promptTokens).toBe(20)
    expect(call.completionTokens).toBe(8)
  })
})
```

<objective>
Foundation for Phase 22: introduce the `environments` PostgreSQL table, a `getDockerClient(envId)` factory, and thread an optional `environmentId` parameter through every existing `docker.*` tRPC route. After this plan, the backend can talk to multiple Docker hosts (socket + tcp-tls); UI selector and outbound agent come in 22-02 and 22-03 respectively.

Purpose: Satisfies MH-01 (environments table) and MH-02 (factory + per-route envId). Without this plan, plans 22-02 and 22-03 have nothing to bind their selector / agent registration to.
Output: Multi-environment-aware backend that is byte-for-byte backwards compatible — existing UI calls (no environmentId) still hit the local socket via the auto-seeded `local` environment.
</objective>

    ```typescript
    import Dockerode from 'dockerode'
    import {getEnvironment, type Environment} from './environments.js'

    // Cache by env.id (string). Key 'local' is canonicalized to LOCAL_ENV_ID before lookup.
    const clientCache = new Map<string, Dockerode>()

    function buildClient(env: Environment): Dockerode {
      switch (env.type) {
        case 'socket':
          return new Dockerode({socketPath: env.socketPath ?? '/var/run/docker.sock'})
        case 'tcp-tls':
          if (!env.tcpHost || !env.tcpPort || !env.tlsCaPem || !env.tlsCertPem || !env.tlsKeyPem) {
            throw new Error(`[env-misconfigured] tcp-tls env '${env.name}' missing connection fields`)
          }
          return new Dockerode({
            host: env.tcpHost,
            port: env.tcpPort,
            protocol: 'https',
            ca: env.tlsCaPem,
            cert: env.tlsCertPem,
            key: env.tlsKeyPem,
          })
        case 'agent':
          // Plan 22-03 will replace this with AgentDockerClient (a Dockerode-shaped wrapper that proxies over WS).
          throw new Error(`[agent-not-implemented] env '${env.name}' is an agent env — agent transport ships in Plan 22-03`)
      }
    }

    export async function getDockerClient(envIdOrAlias: string | null | undefined): Promise<Dockerode> {
      const env = await getEnvironment(envIdOrAlias)
      if (!env) throw new Error(`[env-not-found] environment ${envIdOrAlias ?? '<null>'} not found`)
      const cached = clientCache.get(env.id)
      if (cached) return cached
      const client = buildClient(env)
      clientCache.set(env.id, client)
      return client
    }

    export function invalidateClient(envId: string): void {
      clientCache.delete(envId)
    }

    export function clearAllClients(): void {
      clientCache.clear()
    }
    ```

    The integration test mounts the broker router on a fresh `express()` app, intercepts the upstream `fetch` to /api/agent/stream (the broker proxies to it), and asserts end-to-end behavior for sync + SSE + error paths.

    Mocking strategy: monkey-patch global `fetch` with a Vitest-free approach (since we're using tsx + assert/strict). Save original, replace with a stub that returns a hand-crafted SSE stream, restore at end.

    Skeleton:
    ```typescript
    import assert from 'node:assert/strict'
    import express from 'express'
    import http from 'node:http'
    import {createBrokerRouter} from './router.js'

    /**
     * Minimal fake Livinityd for the broker router's deps. Only fields the broker touches.
     */
    function makeFakeLivinityd(opts: {multiUser: boolean; users: Array<{id: string; isAdmin?: boolean}>; dataDir?: string}) {
      const usersById = new Map(opts.users.map((u) => [u.id, u]))
      const admin = opts.users.find((u) => u.isAdmin)
      return {
        dataDirectory: opts.dataDir || '/tmp/livos-test',
        logger: {log: () => {}, verbose: () => {}, error: () => {}},
        ai: {redis: {get: async (k: string) => k === 'livos:system:multi_user' ? (opts.multiUser ? 'true' : 'false') : null}},
        users: {
          getById: async (id: string) => usersById.get(id) ?? null,
          getAdmin: async () => admin ?? null,
        },
      } as any
    }

    /**
     * Replace global fetch with a stub returning a fake /api/agent/stream SSE stream.
     */
    function mockUpstreamSse(events: Array<{type: string; data?: unknown; turn?: number}>): () => void {
      const original = globalThis.fetch
      globalThis.fetch = (async () => {
        const lines = events.map((e) => `data: ${JSON.stringify(e)}\n\n`).join('')
        const stream = new ReadableStream({
          start(controller) {
            controller.enqueue(new TextEncoder().encode(lines))
            controller.close()
          },
        })
        return new Response(stream, {status: 200, headers: {'Content-Type': 'text/event-stream'}})
      }) as any
      return () => { globalThis.fetch = original }
    }

    /** Spin up an Express app with the broker mounted, return base URL + close fn. */
    function startBrokerApp(livinityd: any): Promise<{url: string; close: () => Promise<void>}> {
      return new Promise((resolve) => {
        const app = express()
        app.use('/u', createBrokerRouter({livinityd}))
        const server = http.createServer(app)
        server.listen(0, '127.0.0.1', () => {
          const addr = server.address() as any
          resolve({
            url: `http://127.0.0.1:${addr.port}`,
            close: () => new Promise((r) => server.close(() => r())),
          })
        })
      })
    }

    async function runTests() {
      const livinityd = makeFakeLivinityd({
        multiUser: true,
        users: [{id: 'admin-1', isAdmin: true}, {id: 'user-2'}],
      })

      // Test 1: sync POST returns Anthropic Messages JSON
      {
        const restore = mockUpstreamSse([
          {type: 'thinking', turn: 1},
          {type: 'chunk', turn: 1, data: 'Hello '},
          {type: 'chunk', turn: 1, data: 'world.'},
          {type: 'final_answer', turn: 1, data: 'Hello world.'},
          {type: 'done', data: {success: true, answer: 'Hello world.', turns: 1, stoppedReason: 'complete'}},
        ])
        const {url, close} = await startBrokerApp(livinityd)
        const res = await fetch(`${url}/u/admin-1/v1/messages`, {
          method: 'POST',
          headers: {'content-type': 'application/json'},
          body: JSON.stringify({model: 'claude-sonnet-4-6', messages: [{role: 'user', content: 'hi'}]}),
        })
        assert.equal(res.status, 200)
        const body = await res.json() as any
        assert.equal(body.type, 'message')
        assert.equal(body.role, 'assistant')
        assert.ok(Array.isArray(body.content) && body.content[0]?.type === 'text')
        assert.match(body.content[0].text, /Hello world\./)
        assert.equal(body.stop_reason, 'end_turn')
        await close()
        restore()
        console.log('  PASS Test 1: sync POST → Anthropic Messages JSON shape')
      }

      // Test 2: SSE POST returns spec-compliant chunks ending in message_stop
      {
        const restore = mockUpstreamSse([
          {type: 'thinking', turn: 1},
          {type: 'chunk', turn: 1, data: 'streamed!'},
          {type: 'final_answer', turn: 1, data: 'streamed!'},
          {type: 'done', data: {success: true, answer: 'streamed!', turns: 1, stoppedReason: 'complete'}},
        ])
        const {url, close} = await startBrokerApp(livinityd)
        const res = await fetch(`${url}/u/admin-1/v1/messages`, {
          method: 'POST',
          headers: {'content-type': 'application/json'},
          body: JSON.stringify({model: 'claude-sonnet-4-6', messages: [{role: 'user', content: 'hi'}], stream: true}),
        })
        assert.equal(res.status, 200)
        assert.equal(res.headers.get('content-type'), 'text/event-stream')
        const text = await res.text()
        assert.match(text, /event: message_start\ndata: /)
        assert.match(text, /event: content_block_start\ndata: /)
        assert.match(text, /event: content_block_delta\ndata: .*streamed!/)
        assert.match(text, /event: message_stop\ndata: /)
        await close()
        restore()
        console.log('  PASS Test 2: SSE POST → spec-compliant chunks')
      }

      // Test 3: unknown userId → 404
      {
        const {url, close} = await startBrokerApp(livinityd)
        const res = await fetch(`${url}/u/nonexistent-user/v1/messages`, {
          method: 'POST', headers: {'content-type': 'application/json'},
          body: JSON.stringify({model: 'x', messages: [{role: 'user', content: 'hi'}]}),
        })
        assert.equal(res.status, 404)
        await close()
        console.log('  PASS Test 3: unknown userId → 404')
      }

      // Test 4: single-user mode + non-admin userId → 403
      {
        const singleUserLivinityd = makeFakeLivinityd({
          multiUser: false,
          users: [{id: 'admin-1', isAdmin: true}, {id: 'user-2'}],
        })
        const {url, close} = await startBrokerApp(singleUserLivinityd)
        const res = await fetch(`${url}/u/user-2/v1/messages`, {
          method: 'POST', headers: {'content-type': 'application/json'},
          body: JSON.stringify({model: 'x', messages: [{role: 'user', content: 'hi'}]}),
        })
        assert.equal(res.status, 403)
        await close()
        console.log('  PASS Test 4: single-user mode + non-admin → 403')
      }

        console.log('  PASS Test 5: invalid body shape → 400')
      }

      console.log('\nAll integration.test.ts tests passed (5/5)')
    }
    runTests().catch(err => { console.error('FAILED:', err); process.exit(1) })
    ```

    The nexus-side change in Plan 41-04 added header-conditional homeOverride. Test it directly without spinning up the full /api/agent/stream handler (which has many dependencies — brain, toolRegistry, daemon, redis).

    Strategy: extract the header-reading + path-computing logic into a verifiable form. Two approaches:

    **Approach A (recommended)**: Test the SdkAgentRunner home-override behavior end-to-end via a minimal handler stub. Create a tiny isolated function that simulates the relevant code-path:
    ```typescript
    import assert from 'node:assert/strict'
    import path from 'node:path'

    /**
     * Mirror of the Plan 41-04 logic in nexus/packages/core/src/api.ts.
     * Tests the regression invariants in isolation.
     *
     * If the real code in api.ts diverges from this mirror, this test will pass
     * but production will be broken — to mitigate, the test file's docstring
     * directs reviewers to keep the two in sync, AND the integration test
     * (livos/.../integration.test.ts Test 1+2) implicitly verifies the wire
     * end-to-end.
     */
    function computeHomeOverride(headerVal: string | string[] | undefined, dataDir = '/opt/livos/data'): string | undefined {
      if (typeof headerVal !== 'string') return undefined
      if (!/^[a-zA-Z0-9_-]+$/.test(headerVal)) return undefined
      return path.join(dataDir, 'users', headerVal, '.claude')
    }

    async function runTests() {
      // Test 1: valid header → homeOverride set
      const r1 = computeHomeOverride('user-abc-123')
      assert.equal(r1, path.join('/opt/livos/data', 'users', 'user-abc-123', '.claude'))
      console.log('  PASS Test 1: valid header → homeOverride set')

      // Test 2: header absent → homeOverride undefined (single-user-mode preservation)
      const r2 = computeHomeOverride(undefined)
      assert.equal(r2, undefined)
      console.log('  PASS Test 2: absent header → homeOverride undefined (byte-identical pre-Phase-41)')

      // Test 3: array-valued header (Express normalizes some headers to arrays) → ignored
      const r3 = computeHomeOverride(['user-1', 'user-2'])
      assert.equal(r3, undefined)
      console.log('  PASS Test 3: array-valued header rejected')

      // Test 4: malformed header (path traversal attempt) → undefined
      const r4 = computeHomeOverride('../../../etc/passwd')
      assert.equal(r4, undefined)
      console.log('  PASS Test 4: path-traversal header rejected')

      // Test 5: empty string → undefined
      const r5 = computeHomeOverride('')
      assert.equal(r5, undefined)
      console.log('  PASS Test 5: empty header rejected')

      // Test 6: env var override
      const r6 = computeHomeOverride('user-1', '/custom/data')
      assert.equal(r6, '/custom/data/users/user-1/.claude')
      console.log('  PASS Test 6: LIVOS_DATA_DIR override honored')

      // Test 7: api.ts source contains the same logic (regression check via grep)
      const fs = await import('node:fs/promises')
      const apiSrc = await fs.readFile(new URL('../api.ts', import.meta.url), 'utf8')
      assert.match(apiSrc, /x-livos-user-id/i, 'api.ts must read X-LivOS-User-Id header')
      assert.match(apiSrc, /homeOverride/, 'api.ts must set homeOverride')
      assert.match(apiSrc, /LIVOS_DATA_DIR/, 'api.ts must read LIVOS_DATA_DIR env var')
      assert.match(apiSrc, /\^\[a-zA-Z0-9_-\]\+\$/, 'api.ts must apply userId regex validation')
      console.log('  PASS Test 7: api.ts source contains all required Phase 41 wiring')

      console.log('\nAll api-home-override.test.ts tests passed (7/7)')
    }
    runTests().catch(err => { console.error('FAILED:', err); process.exit(1) })
    ```

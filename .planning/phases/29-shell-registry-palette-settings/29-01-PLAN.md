       ```typescript
       const KEY = 'livos:docker:palette:recent'
       const MAX = 8
       export function useRecentSearches() {
         const [recent, setRecent] = useState<string[]>(() => {
           try { return JSON.parse(localStorage.getItem(KEY) ?? '[]') } catch { return [] }
         })
         const addRecent = useCallback((q: string) => {
           const trimmed = q.trim()
           if (!trimmed) return
           setRecent((prev) => {
             const next = [trimmed, ...prev.filter((p) => p !== trimmed)].slice(0, MAX)
             localStorage.setItem(KEY, JSON.stringify(next))
             return next
           })
         }, [])
         return {recent, addRecent}
       }
       ```

    **Anti-pitfall — input event vs keydown for cmd+k:** Always use `keydown` listener on `document` (not `keypress`/`keyup`). Always call `e.preventDefault()` AFTER the metaKey+ctrlKey check so non-cmd+k keystrokes pass through. Common gotcha: Firefox sometimes triggers cmd+k as the search bar; preventDefault is required to stop that.

    **Anti-pitfall — recent searches store query, not result:** addRecent(query) is called with the typed query, NOT the selected result label. This matches the user's mental model ("I just searched for foo" not "I just opened the foo container").

Order: `addRecent(query); closePalette(); setSelectedX(id); setSection(...)`. The set* calls trigger a re-render of the page underneath; the modal is already gone.

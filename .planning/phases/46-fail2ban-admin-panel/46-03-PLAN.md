```typescript
// fail2ban-admin/routes.ts
import { z } from 'zod';
import { TRPCError } from '@trpc/server';
import { adminProcedure, router } from '../server/trpc/trpc.js';
import {
  listJails as listJailsImpl,
  getJailStatus as getJailStatusImpl,
  unbanIp as unbanIpImpl,
  banIp as banIpImpl,
  addIgnoreIp as addIgnoreIpImpl,
  listEvents as listEventsImpl,
  Fail2banClientError,
  realActiveSessionsProvider,
  recordFail2banEvent,
} from './index.js';

const ipSchema = z.string()
  .trim()
  .refine(
    (s) => /^(?:25[0-5]|2[0-4]\d|[01]?\d\d?)(?:\.(?:25[0-5]|2[0-4]\d|[01]?\d\d?)){3}$/.test(s),
    'must be IPv4 dotted-quad (no CIDR allowed in v29.4 — pitfall B-03)'
  );

const jailSchema = z.string()
  .trim()
  .regex(/^[a-zA-Z0-9_.-]+$/, 'jail name must match [a-zA-Z0-9_.-]+');

const fail2banRouter = router({
  listJails: adminProcedure.query(async () => { ... }),
  getJailStatus: adminProcedure.input(z.object({ jail: jailSchema })).query(...),
  listEvents: adminProcedure.input(z.object({ limit: z.number().int().min(1).max(200).default(50) })).query(...),
  unbanIp: adminProcedure
    .input(z.object({ jail: jailSchema, ip: ipSchema, addToWhitelist: z.boolean().default(false) }))
    .mutation(...),
  banIp: adminProcedure
    .input(z.object({
      jail: jailSchema,
      ip: ipSchema,
      confirmation: z.literal('LOCK ME OUT').optional(),
      cellularBypass: z.boolean().default(false),
    }))
    .mutation(...),
});

export default fail2banRouter;
```

<objective>
Enable the Kimi AI provider to see images (vision) and fix the native tool calling path so that tool result images (screenshots from device tools) are passed through to the LLM instead of being silently dropped.

Purpose: Without this, the AI cannot see screenshots. The existing architecture carries images through ChatMessage -> normalizeMessages -> prepareForProvider -> image_url, but the native tool calling path (used by Kimi) builds ToolResultBlock with `content: string` only, dropping any images from ToolResult.images. This plan fixes that gap and enables vision.

Output: Kimi processes images, tool result screenshots flow to the LLM in native mode
</objective>

const resultText = toolResult.success
  ? toolResult.output

// Build content: if tool returned images, create multimodal content array
let resultContent: string | Array<{ type: string; [k: string]: unknown }> = resultText;
if (toolResult.images && toolResult.images.length > 0) {
  resultContent = [
    { type: 'text', text: resultText },
    ...toolResult.images.map(img => ({
      type: 'image_url' as const,
      image_url: { url: `data:${img.mimeType};base64,${img.base64}` },
    })),
  ];
}

toolResultBlocks.push({
  type: 'tool_result',
  tool_use_id: toolCall.id,
  content: resultContent,
  is_error: !toolResult.success,
});

    import {spawn as realSpawn} from 'node:child_process'

    export interface SshSessionEvent {
      // Microseconds-since-epoch as a string (matches journalctl's __REALTIME_TIMESTAMP wire format).
      // UI converts via Number(ts) / 1000 for ms-precision Date.
      timestamp: string
      message: string
      ip: string | null  // null when MESSAGE has no "from <ip>" pattern (e.g., pam session lines)
      hostname?: string
    }

    export interface SpawnFn {
      (binary: string, args: string[]): {
        stdout: NodeJS.ReadableStream
        stderr: NodeJS.ReadableStream
        on(event: 'error' | 'exit' | 'close', listener: (...args: any[]) => void): void
        kill(signal?: string): void
      }
    }

    export interface JournalctlStream {
      // Per-connection subscribe — caller (ws-handler) attaches one listener.
      subscribe(listener: (ev: SshSessionEvent) => void): () => void  // returns unsubscribe
      // Stop the spawned process — called when the last subscriber disconnects.
      stop(): void
    }

    export interface MakeStreamOptions {
      onMissing?: () => void   // ENOENT — journalctl binary missing; ws-handler closes WS with 4404
      onExit?: (code: number | null, signal: string | null) => void
      logger?: {warn: (...a: unknown[]) => void; error?: (...a: unknown[]) => void}
    }

    // Pure helper — exported so journalctl-stream.test.ts can drive it directly.
    // Regex MUST match both Failed-password and Accepted-publickey forms (and any other
    // "from <ip>" pattern that sshd emits — e.g., Disconnected, Invalid user). Returns
    // null when no match. IPv4 only for v29.4 (IPv6 deferred — same scope as Phase 46
    // ipSchema which is also IPv4-only — no `maxmind`).
    export function extractIp(message: string): string | null {
      const m = message.match(/\bfrom\s+(\d{1,3}(?:\.\d{1,3}){3})\b/)
      return m ? m[1] : null
    }

    export function makeJournalctlStream(spawn: SpawnFn, opts: MakeStreamOptions = {}): JournalctlStream {
      const listeners = new Set<(ev: SshSessionEvent) => void>()
      let buffer = ''
      let stopped = false
      let child: ReturnType<SpawnFn> | null = null

      try {
        child = spawn('journalctl', ['-u', 'ssh', '-o', 'json', '--follow', '--since', '1 hour ago'])
      } catch (err) {
        // synchronous spawn failure (extremely rare — usually emits 'error' event instead)
        const code = (err as {code?: string})?.code ?? ''
        if (code === 'ENOENT') opts.onMissing?.()
        opts.logger?.warn?.('[ssh-sessions] spawn failed:', (err as Error)?.message || err)
        return makeNoOpStream()
      }

      child.on('error', (err: NodeJS.ErrnoException) => {
        if (err?.code === 'ENOENT') {
          opts.onMissing?.()
          return
        }
        opts.logger?.warn?.('[ssh-sessions] journalctl error:', err?.message || err)
      })

      child.on('exit', (code: number | null, signal: string | null) => {
        opts.onExit?.(code, signal)
      })

      child.stdout.on('data', (chunk: Buffer | string) => {
        buffer += typeof chunk === 'string' ? chunk : chunk.toString('utf8')
        let nl: number
        while ((nl = buffer.indexOf('\n')) !== -1) {
          const line = buffer.slice(0, nl).trim()
          buffer = buffer.slice(nl + 1)
          if (!line) continue
          let parsed: Record<string, unknown>
          try {
            parsed = JSON.parse(line)
          } catch {
            continue  // malformed mid-flush — skip silently
          }
          // Defensive filter even though `-u ssh` already gates at journalctl level.
          if (parsed._SYSTEMD_UNIT && parsed._SYSTEMD_UNIT !== 'ssh.service') continue
          const message = typeof parsed.MESSAGE === 'string' ? parsed.MESSAGE : ''
          const ts = typeof parsed.__REALTIME_TIMESTAMP === 'string' ? parsed.__REALTIME_TIMESTAMP : String(Date.now() * 1000)
          const ev: SshSessionEvent = {
            timestamp: ts,
            message,
            ip: extractIp(message),
            hostname: typeof parsed._HOSTNAME === 'string' ? parsed._HOSTNAME : undefined,
          }
          for (const l of listeners) {
            try { l(ev) } catch (lerr) { opts.logger?.warn?.('[ssh-sessions] listener throw:', lerr) }
          }
        }
      })

      return {
        subscribe(listener) {
          listeners.add(listener)
          return () => { listeners.delete(listener) }
        },
        stop() {
          if (stopped) return
          stopped = true
          listeners.clear()
          try { child?.kill('SIGTERM') } catch {}
        },
      }

      function makeNoOpStream(): JournalctlStream {
        return {subscribe: () => () => {}, stop: () => {}}
      }
    }

    // Production-wired stream factory. Uses real node:child_process.spawn.
    // Caller passes onMissing callback to handle ENOENT (binary missing on host) gracefully.
    export function realJournalctlStream(opts: MakeStreamOptions = {}): JournalctlStream {
      const adapter: SpawnFn = (binary, args) => {
        const c = realSpawn(binary, args, {stdio: ['ignore', 'pipe', 'pipe']})
        return {
          stdout: c.stdout!,
          stderr: c.stderr!,
          on: c.on.bind(c),
          kill: (sig?: string) => { c.kill(sig as NodeJS.Signals | undefined) },
        }
      }
      return makeJournalctlStream(adapter, opts)
    }

     * Phase 48 Plan 48-01 — WebSocket handler for /ws/ssh-sessions.
     *
     * URL: `/ws/ssh-sessions?token=<jwt>`
     *
     * RBAC gate: re-verifies the token at the handler boundary (the generic
     * mountWebSocketServer flow at server/index.ts:905-934 only checks token
     * validity — NOT user role). Mirrors the admin-role lookup pattern from
     * server/index.ts:1170-1200 (/api/files privateApi middleware): decode JWT
     * payload.userId, look up via findUserById, gate on role === 'admin'.
     *
     * Close codes:
     *   4403 — non-admin user (mirror v26.0 authorizeDeviceAccess close-code
     *           convention; 4xxx range = application-defined per RFC 6455).
     *   4404 — `journalctl` binary missing on host (graceful degrade — Mini
     *           PC always has it, but defensive).
     *   1011 — internal handler error (RFC 6455 standard "server error").
     *
     * Ring buffer: 5000 events (mirror Phase 28 docker-logs MAX_LINES_PER_CONTAINER).
     * Per-connection — when a client connects, handler replays the last N events
     * from the buffer immediately, then live-tails new events. (Optional Plan 48-02
     * may opt out of replay; this plan ships the buffer + replay-on-connect.)
     *
     * D-NO-NEW-DEPS upheld — no maxmind, no GeoIP, no third-party JWT library;
     * uses livinityd's existing this.verifyToken (jwt.verify) + findUserById.
     */

    import type {WebSocket} from 'ws'
    import type http from 'node:http'
    import type Livinityd from '../../index.js'
    import {makeJournalctlStream, realJournalctlStream, type JournalctlStream, type SshSessionEvent} from './journalctl-stream.js'
    import type {SpawnFn} from './journalctl-stream.js'

    const RING_BUFFER_LIMIT = 5_000

    interface CreateHandlerDeps {
      livinityd: Livinityd
      logger: {warn: (...a: unknown[]) => void; error: (...a: unknown[]) => void; verbose?: (...a: unknown[]) => void}
      // For tests — inject a streamFactory so unit tests don't spawn real journalctl.
      streamFactory?: (opts: {onMissing: () => void; onExit?: (c: number | null, s: string | null) => void; logger: {warn: (...a: unknown[]) => void}}) => JournalctlStream
    }

    interface SharedStreamState {
      stream: JournalctlStream
      buffer: SshSessionEvent[]   // ring (most-recent N)
      subscribers: Set<(ev: SshSessionEvent) => void>
    }

    export function createSshSessionsWsHandler(deps: CreateHandlerDeps) {
      let shared: SharedStreamState | null = null

      function ensureStream(onMissing: () => void): SharedStreamState {
        if (shared) return shared
        const buffer: SshSessionEvent[] = []
        const subscribers = new Set<(ev: SshSessionEvent) => void>()
        const factory = deps.streamFactory ?? realJournalctlStream
        const stream = factory({
          onMissing,
          onExit: () => { shared = null },  // process died — next connection re-spawns
          logger: deps.logger,
        })
        // Fan-out subscriber that pushes into ring buffer + every per-WS subscriber.
        stream.subscribe((ev) => {
          buffer.push(ev)
          if (buffer.length > RING_BUFFER_LIMIT) buffer.shift()  // drop oldest — pitfall: O(n) shift, but 5000 is fine
          for (const sub of subscribers) {
            try { sub(ev) } catch (err) { deps.logger.warn('[ssh-sessions] subscriber threw:', err) }
          }
        })
        shared = {stream, buffer, subscribers}
        return shared
      }

      return async function (ws: WebSocket, request: http.IncomingMessage) {
        // 1. Re-verify JWT and resolve admin role.
        let userId: string | null = null
        try {
          const url = new URL(`https://localhost${request.url ?? '/'}`)
          const token = url.searchParams.get('token')
          if (!token) { ws.close(4403, 'missing token'); return }
          const payload = (await deps.livinityd.verifyToken(token)) as {userId?: string; loggedIn?: boolean}
          // Legacy tokens have {loggedIn: true} but no userId — those map to admin via getAdminUser().
          // New multi-user tokens have userId. Accept either path BUT require role === 'admin' below.
          if (typeof payload.userId === 'string') {
            userId = payload.userId
          } else if (payload.loggedIn === true) {
            // Legacy single-user mode — admin by definition. Allow.
            // (Same fallback used by ai/routes.ts is-authenticated middleware.)
            const {findUserById, getAdminUser} = await import('../database/index.js')
            const admin = await getAdminUser()
            if (!admin) { ws.close(4403, 'admin role required'); return }
            userId = admin.id
          } else {
            ws.close(4403, 'admin role required'); return
          }
        } catch (err) {
          deps.logger.warn('[ssh-sessions] token verify failed:', (err as Error)?.message || err)
          ws.close(4403, 'admin role required'); return
        }

        // 2. Look up user role.
        try {
          const {findUserById} = await import('../database/index.js')
          const user = await findUserById(userId)
          if (!user || user.role !== 'admin') {
            ws.close(4403, 'admin role required'); return
          }
        } catch (err) {
          deps.logger.error('[ssh-sessions] role lookup failed:', err)
          ws.close(1011, 'server error'); return
        }

        // 3. Ensure journalctl stream — close 4404 on ENOENT.
        let missingClosed = false
        const state = ensureStream(() => {
          if (!missingClosed) {
            missingClosed = true
            try { ws.close(4404, 'journalctl binary missing on host') } catch {}
          }
        })
        if (missingClosed) return  // close already fired synchronously

        // 4. Replay ring-buffer (most-recent N) on connect.
        for (const ev of state.buffer) {
          if (ws.readyState !== ws.OPEN) break
          try { ws.send(JSON.stringify(ev)) } catch {}
        }

        // 5. Live subscribe.
        const sub = (ev: SshSessionEvent) => {
          if (ws.readyState !== ws.OPEN) return
          try { ws.send(JSON.stringify(ev)) } catch (err) {
            deps.logger.warn('[ssh-sessions] ws.send failed:', err)
          }
        }
        state.subscribers.add(sub)

        // 6. Cleanup on close.
        ws.on('close', () => {
          state.subscribers.delete(sub)
          // If we were the last subscriber, tear down the spawned process.
          if (state.subscribers.size === 0) {
            try { state.stream.stop() } catch {}
            shared = null
          }
        })
        ws.on('error', (err) => {
          deps.logger.warn('[ssh-sessions] ws error:', err?.message || err)
        })
      }
    }

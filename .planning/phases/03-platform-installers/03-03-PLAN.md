**2. Create agent/installer/linux/livinity-agent.service** -- systemd user service:
```ini
[Unit]
Description=Livinity Agent - Remote PC Control
Documentation=https://livinity.io
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
ExecStart=/usr/local/bin/livinity-agent start --background
ExecStop=/usr/local/bin/livinity-agent stop
Restart=on-failure
RestartSec=10
Environment=HOME=%h
WorkingDirectory=%h

[Install]
WantedBy=default.target
```

Note: This is a **user** service (installed to `~/.config/systemd/user/`), not a system service. Using `%h` for home directory. `WantedBy=default.target` is the user-session equivalent of `multi-user.target`. This ensures the service runs as the installing user (LIN-03) without needing root after initial package install.

Actually, since the .deb installs a system-wide binary at `/usr/local/bin/`, the systemd service can be a system service with `User=` directive set at install time. But the CONTEXT.md says "Service runs as the installing user (User= directive, detected at install time)". So we need a **system** service with dynamic User= detection.

Revised systemd service (system service with placeholder):
```ini
[Unit]
Description=Livinity Agent - Remote PC Control
Documentation=https://livinity.io
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
ExecStart=/usr/local/bin/livinity-agent start --background
ExecStop=/usr/local/bin/livinity-agent stop
Restart=on-failure
RestartSec=10
User=__USER__
Group=__USER__
Environment=HOME=/home/__USER__
WorkingDirectory=/home/__USER__

[Install]
WantedBy=multi-user.target
```

The `__USER__` placeholder gets replaced by postinst.sh with the SUDO_USER or current user.

**3. Create agent/installer/linux/postinst.sh** -- post-install script:
```bash
#!/bin/bash
set -e

# Detect the installing user (not root)
INSTALL_USER="${SUDO_USER:-$(logname 2>/dev/null || whoami)}"
INSTALL_HOME=$(eval echo "~$INSTALL_USER")

SERVICE_FILE="/etc/systemd/system/livinity-agent.service"

# Replace user placeholder in service file
if [ -f "$SERVICE_FILE" ]; then
    sed -i "s/__USER__/$INSTALL_USER/g" "$SERVICE_FILE"
    sed -i "s|/home/__USER__|$INSTALL_HOME|g" "$SERVICE_FILE"
fi

# Create credentials directory
mkdir -p "$INSTALL_HOME/.livinity"
chown "$INSTALL_USER:$INSTALL_USER" "$INSTALL_HOME/.livinity"

# Reload systemd and enable service
systemctl daemon-reload
systemctl enable livinity-agent.service
systemctl start livinity-agent.service || true

echo ""
echo "Livinity Agent installed successfully!"
echo "  Binary: /usr/local/bin/livinity-agent"
echo "  Service: livinity-agent.service (running as $INSTALL_USER)"
echo "  Config: $INSTALL_HOME/.livinity/"
echo ""
echo "The agent will start automatically on boot."
echo "Run 'livinity-agent status' to check connection status."
```

**4. Create agent/installer/linux/prerm.sh** -- pre-remove script:
```bash
#!/bin/bash
set -e

# Stop and disable service before removal
if systemctl is-active --quiet livinity-agent.service 2>/dev/null; then
    systemctl stop livinity-agent.service || true
fi
if systemctl is-enabled --quiet livinity-agent.service 2>/dev/null; then
    systemctl disable livinity-agent.service || true
fi
systemctl daemon-reload || true
```

**5. Create agent/installer/linux/build-deb.sh** -- build script using fpm:
```bash
#!/bin/bash
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
AGENT_DIR="$(cd "$SCRIPT_DIR/../.." && pwd)"
DIST_DIR="$AGENT_DIR/dist"
VERSION="0.1.0"
ARCH="amd64"
PKG_NAME="livinity-agent"
OUTPUT_DIR="$DIST_DIR/installer"

echo "=== Building Livinity Agent Linux .deb Package ==="
echo ""

# Step 1: Build SEA binary
echo "Step 1: Building SEA binary..."
cd "$AGENT_DIR"
npm run build:sea
echo ""

# Step 2: Check for fpm
if ! command -v fpm &>/dev/null; then
    echo "ERROR: fpm not found. Install it with: gem install fpm"
    echo "  or: sudo apt install ruby-dev build-essential && sudo gem install fpm"
    exit 1
fi

# Step 3: Create staging directory
echo "Step 2: Creating package structure..."
STAGING="$DIST_DIR/deb-staging"
rm -rf "$STAGING"

# Binary
mkdir -p "$STAGING/usr/local/bin"
cp "$DIST_DIR/livinity-agent" "$STAGING/usr/local/bin/livinity-agent"
chmod +x "$STAGING/usr/local/bin/livinity-agent"

# Native dependencies (systray2 tray binary)
mkdir -p "$STAGING/usr/local/lib/livinity-agent/node_modules/systray2/traybin"
cp -R "$DIST_DIR/node_modules/systray2"/* "$STAGING/usr/local/lib/livinity-agent/node_modules/systray2/"

# node-screenshots native module (if present for Linux)
if [ -d "$DIST_DIR/node_modules/node-screenshots" ]; then
    mkdir -p "$STAGING/usr/local/lib/livinity-agent/node_modules/node-screenshots"
    cp -R "$DIST_DIR/node_modules/node-screenshots"/* "$STAGING/usr/local/lib/livinity-agent/node_modules/node-screenshots/"
fi
if [ -d "$DIST_DIR/node_modules/@node-rs" ]; then
    mkdir -p "$STAGING/usr/local/lib/livinity-agent/node_modules/@node-rs"
    cp -R "$DIST_DIR/node_modules/@node-rs"/* "$STAGING/usr/local/lib/livinity-agent/node_modules/@node-rs/"
fi

# Setup UI
mkdir -p "$STAGING/usr/local/lib/livinity-agent/setup-ui"
cp -R "$DIST_DIR/setup-ui"/* "$STAGING/usr/local/lib/livinity-agent/setup-ui/"

# systemd service file (with placeholder -- postinst replaces __USER__)
mkdir -p "$STAGING/etc/systemd/system"
cp "$SCRIPT_DIR/livinity-agent.service" "$STAGING/etc/systemd/system/livinity-agent.service"

# Step 4: Build .deb with fpm
echo "Step 3: Building .deb package..."
mkdir -p "$OUTPUT_DIR"

fpm \
    --input-type dir \
    --output-type deb \
    --name "$PKG_NAME" \
    --version "$VERSION" \
    --architecture "$ARCH" \
    --description "Livinity Agent - Remote PC control via livinity.io" \
    --url "https://livinity.io" \
    --maintainer "Livinity <support@livinity.io>" \
    --after-install "$SCRIPT_DIR/postinst.sh" \
    --before-remove "$SCRIPT_DIR/prerm.sh" \
    --package "$OUTPUT_DIR/${PKG_NAME}_${VERSION}_${ARCH}.deb" \
    --force \
    --chdir "$STAGING" \
    .

# Clean up staging
rm -rf "$STAGING"

echo ""
echo "SUCCESS: .deb package created at $OUTPUT_DIR/${PKG_NAME}_${VERSION}_${ARCH}.deb"
echo "  Install with: sudo dpkg -i ${PKG_NAME}_${VERSION}_${ARCH}.deb"
```

Make the script executable: `chmod +x agent/installer/linux/build-deb.sh`

**1. Add installSystemdService() function to agent/src/cli.ts:**

Add a new function that programmatically installs a systemd **user** service (no root needed) when running on Linux outside of a .deb package context. This handles the case where a user runs the SEA binary directly without the .deb.

```typescript
function installSystemdService(): void {
  if (process.platform !== 'linux') return;

  const home = os.homedir();
  const serviceDir = path.join(home, '.config', 'systemd', 'user');
  const servicePath = path.join(serviceDir, 'livinity-agent.service');

  // Skip if already installed
  if (fs.existsSync(servicePath)) return;

  // Skip if running inside a .deb install (system service exists)
  if (fs.existsSync('/etc/systemd/system/livinity-agent.service')) return;

  const agentPath = process.execPath;
  const serviceContent = `[Unit]
Description=Livinity Agent - Remote PC Control
Documentation=https://livinity.io
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
ExecStart=${agentPath} start --background
ExecStop=${agentPath} stop
Restart=on-failure
RestartSec=10

[Install]
WantedBy=default.target
`;

  try {
    fs.mkdirSync(serviceDir, { recursive: true });
    fs.writeFileSync(servicePath, serviceContent, 'utf-8');
    console.log('[agent] Installed systemd user service for auto-start');
    console.log('[agent] Enable with: systemctl --user enable livinity-agent && systemctl --user start livinity-agent');
  } catch (err) {
    // Non-fatal -- skip if can't write
    console.warn('[agent] Could not install systemd user service:', (err as Error).message);
  }
}
```

Import `fs` and `path` at top of cli.ts if not already imported (they likely are via state.ts usage, but add explicit imports if needed: `import fs from 'node:fs'` and `import path from 'node:path'`).

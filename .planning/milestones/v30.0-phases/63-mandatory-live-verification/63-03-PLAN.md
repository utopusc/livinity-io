```python
#!/usr/bin/env python3
"""
Phase 63 FR-VERIFY-V30-06 — Anthropic Python SDK smoke test.
Mirrors v29.3 Phase 42 UAT pattern.

Usage:
    pip install anthropic
    LIV_SK_PHASE63=liv_sk_xxx python3 sdk-smoke.py
"""

import os
import sys
import json
from anthropic import Anthropic

api_key = os.environ.get("LIV_SK_PHASE63")
if not api_key:
    print("ERROR: LIV_SK_PHASE63 env var not set", file=sys.stderr)
    sys.exit(2)

if not api_key.startswith("liv_sk_"):
    print("ERROR: key does not have liv_sk_ prefix; this is not a Livinity key", file=sys.stderr)
    sys.exit(2)

client = Anthropic(
    api_key=api_key,
    base_url="https://api.livinity.io",
)

print(f"=== sdk-smoke begin ===", flush=True)
print(f"base_url=https://api.livinity.io", flush=True)
print(f"key_prefix={api_key[:12]}...", flush=True)

# Sync request
print("--- sync request ---", flush=True)
resp = client.messages.create(
    model="opus",
    max_tokens=150,
    messages=[
        {"role": "user", "content": "Count from 1 to 5, one number per line."}
    ],
)

print(f"id={resp.id}", flush=True)
print(f"model={resp.model}", flush=True)
print(f"role={resp.role}", flush=True)
print(f"stop_reason={resp.stop_reason}", flush=True)
print(f"usage.input_tokens={resp.usage.input_tokens}", flush=True)
print(f"usage.output_tokens={resp.usage.output_tokens}", flush=True)

# Extract text content
text_parts = []
for block in resp.content:
    if hasattr(block, "text"):
        text_parts.append(block.text)
text = "\n".join(text_parts)
print(f"--- response text ---", flush=True)
print(text, flush=True)
print(f"--- end response text ---", flush=True)

# Assertions
assert resp.id, "id must be non-empty"
assert resp.usage.input_tokens > 0, "input_tokens must be > 0"
assert resp.usage.output_tokens > 0, "output_tokens must be > 0"
assert text.strip(), "response text must be non-empty"

print(f"=== sdk-smoke PASS ===", flush=True)
```

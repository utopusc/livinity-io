```bash
{
  echo "=== PRE-FLIGHT CHECK 1: Phases 56-62 EXECUTED ==="
  date -u +%Y-%m-%dT%H:%M:%SZ
  for phase in 56 57 58 59 60 61 62; do
    dir=$(ls -d .planning/phases/${phase}-* 2>/dev/null | head -1)
    if [ -z "$dir" ]; then
      echo "BLOCK: Phase ${phase} directory does not exist"
      continue
    fi
    summary_count=$(ls "$dir"/*-SUMMARY.md 2>/dev/null | wc -l)
    verify_count=$(ls "$dir"/*-VERIFICATION.md 2>/dev/null | wc -l)
    human_needed=$(grep -l 'status:\s*human_needed' "$dir"/*-VERIFICATION.md 2>/dev/null | wc -l)
    passed=$(grep -l 'status:\s*passed' "$dir"/*-VERIFICATION.md 2>/dev/null | wc -l)
    if [ "$summary_count" -lt 1 ]; then echo "BLOCK: Phase ${phase} has zero *-SUMMARY.md"; continue; fi
    if [ "$verify_count" -lt 1 ]; then echo "BLOCK: Phase ${phase} has zero *-VERIFICATION.md"; continue; fi
    if [ "$human_needed" -gt 0 ]; then echo "BLOCK: Phase ${phase} has ${human_needed} VERIFICATION file(s) with status: human_needed"; continue; fi
    if [ "$passed" -lt 1 ]; then echo "BLOCK: Phase ${phase} has no VERIFICATION with status: passed"; continue; fi
    echo "OK: Phase ${phase} — summaries=${summary_count}, verifications=${verify_count}, passed=${passed}, human_needed=${human_needed}"
  done
  echo "=== END PRE-FLIGHT CHECK 1 ==="
} | tee -a .planning/phases/63-mandatory-live-verification/63-pre-flight.log
```

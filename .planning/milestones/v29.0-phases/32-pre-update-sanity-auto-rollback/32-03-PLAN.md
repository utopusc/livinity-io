```bash
#!/usr/bin/env bash
# Phase 32 patch for /opt/livos/update.sh + systemd:
#   REL-01: precheck() + record_previous_sha() helpers
#   REL-02: livos.service drop-in + livos-rollback.service oneshot + livos-rollback.sh
# Idempotent. Apply via:  ssh <host> 'sudo bash -s' < phase32-systemd-rollback-patch.sh
# Backup safety: writes /opt/livos/update.sh.pre-phase32; bash -n at end with restore-on-fail.

set -euo pipefail

UPDATE_SH="/opt/livos/update.sh"
DROPIN_DIR="/etc/systemd/system/livos.service.d"
DROPIN_PATH="$DROPIN_DIR/auto-rollback.conf"
ROLLBACK_UNIT="/etc/systemd/system/livos-rollback.service"
ROLLBACK_SH_PATH="/opt/livos/livos-rollback.sh"
HISTORY_DIR="/opt/livos/data/update-history"

# These marker constants MUST match the strings in the embedded artifact bodies
# (the actual box-drawing chars present in precheck-block.sh / livos-rollback.sh /
# livos-rollback.service / auto-rollback.conf — copy verbatim).
MARKER_PRECHECK="<COPY VERBATIM from precheck-block.sh: # ── Phase 32 REL-01: precheck ──>"
MARKER_SHA_ROT="<COPY VERBATIM from precheck-block.sh: # ── Phase 32 REL-02 prep: SHA rotation ──>"
MARKER_DROPIN="<COPY VERBATIM from auto-rollback.conf: # ── Phase 32 REL-02: livos.service drop-in ──>"
MARKER_UNIT="<COPY VERBATIM from livos-rollback.service: # ── Phase 32 REL-02: livos-rollback.service oneshot ──>"
MARKER_ROLLBACK="<COPY VERBATIM from livos-rollback.sh: # ── Phase 32 REL-02: livos-rollback.sh ──>"

if [[ ! -f "$UPDATE_SH" ]]; then echo "ERROR: $UPDATE_SH not found" >&2; exit 1; fi

if [[ ! -f "$UPDATE_SH.pre-phase32" ]]; then
    cp "$UPDATE_SH" "$UPDATE_SH.pre-phase32"
    echo "Backup written: $UPDATE_SH.pre-phase32"
else
    echo "Backup already exists: $UPDATE_SH.pre-phase32 (re-run safe)"
fi

mkdir -p "$HISTORY_DIR"

SYSTEMD_VER_RAW=$(systemctl --version 2>/dev/null | head -1 | awk '{print $2}')
SYSTEMD_VER=$(echo "$SYSTEMD_VER_RAW" | sed 's/[^0-9].*//')
SYSTEMD_VER=${SYSTEMD_VER:-0}
INSTALL_DROPIN=true
if (( SYSTEMD_VER < 254 )); then
    echo "WARN: systemd v$SYSTEMD_VER_RAW too old for RestartMode=direct (need >=254)."
    echo "WARN: rollback drop-in will NOT install — OnFailure= would misfire on every restart."
    echo "WARN: precheck patches + livos-rollback.sh + livos-rollback.service will still install."
    INSTALL_DROPIN=false
else
    echo "INFO: systemd v$SYSTEMD_VER_RAW (>= 254) — auto-rollback.conf drop-in will install."
fi

# Patch 1: install precheck + record_previous_sha helpers in update.sh
if grep -qF "$MARKER_PRECHECK" "$UPDATE_SH"; then
    echo "ALREADY-PATCHED (REL-01 helpers): present in update.sh"
else
    ANCHOR_LINE=$(grep -n '^fail()' "$UPDATE_SH" | head -1 | cut -d: -f1)
    [[ -z "$ANCHOR_LINE" ]] && { echo "ERROR: no ^fail() anchor" >&2; exit 1; }
    INSERT_AT=$((ANCHOR_LINE + 1))
    PATCH_BLOCK=$(mktemp)
    cat <<'EOPRECHECK' > "$PATCH_BLOCK"
<<<INLINE: precheck-block.sh — body only. Drop the shebang and long header. Start at the `# ── Phase 32 REL-01: precheck ──` marker line and end at the closing `}` of `record_previous_sha()`.>>>
EOPRECHECK
    awk -v line="$INSERT_AT" -v patchfile="$PATCH_BLOCK" '
        BEGIN { while ((getline pl < patchfile) > 0) patch = patch pl ORS }
        NR == line { printf "%s", patch }
        { print }
    ' "$UPDATE_SH" > "$UPDATE_SH.new"
    mv "$UPDATE_SH.new" "$UPDATE_SH"
    chmod +x "$UPDATE_SH"
    rm -f "$PATCH_BLOCK"
    echo "PATCH-OK (REL-01 helpers): inserted at line $INSERT_AT"
fi

# Patch 2: insert call to precheck after 'ok "Pre-flight passed"'
PRECHECK_CALL_MARKER='# Phase 32 REL-01 call site'
if grep -qF "$PRECHECK_CALL_MARKER" "$UPDATE_SH"; then
    echo "ALREADY-PATCHED (REL-01 call): wired"
else
    PREFLIGHT_LINE=$(grep -n 'ok "Pre-flight passed"' "$UPDATE_SH" | head -1 | cut -d: -f1)
    [[ -z "$PREFLIGHT_LINE" ]] && { echo "ERROR: no Pre-flight anchor" >&2; exit 1; }
    INSERT_AT=$((PREFLIGHT_LINE + 1))
    awk -v line="$INSERT_AT" -v marker="$PRECHECK_CALL_MARKER" '
        NR == line { print ""; print marker; print "precheck" }
        { print }
    ' "$UPDATE_SH" > "$UPDATE_SH.new"
    mv "$UPDATE_SH.new" "$UPDATE_SH"
    chmod +x "$UPDATE_SH"
    echo "PATCH-OK (REL-01 call): inserted after line $PREFLIGHT_LINE"
fi

# Patch 3: insert call to record_previous_sha before SHA-write
SHA_ROT_CALL_MARKER='# Phase 32 REL-02 prep call site'
if grep -qF "$SHA_ROT_CALL_MARKER" "$UPDATE_SH"; then
    echo "ALREADY-PATCHED (REL-02 SHA-rot call): wired"
else
    SHA_LINE=$(grep -n 'git -C "\$TEMP_DIR" rev-parse HEAD > /opt/livos/.deployed-sha' "$UPDATE_SH" | head -1 | cut -d: -f1)
    if [[ -z "$SHA_LINE" ]]; then
        echo "WARN: no SHA-write anchor — record_previous_sha NOT wired"
    else
        awk -v line="$SHA_LINE" -v marker="$SHA_ROT_CALL_MARKER" '
            NR == line { print "    " marker; print "    record_previous_sha" }
            { print }
        ' "$UPDATE_SH" > "$UPDATE_SH.new"
        mv "$UPDATE_SH.new" "$UPDATE_SH"
        chmod +x "$UPDATE_SH"
        echo "PATCH-OK (REL-02 SHA-rot call): inserted before line $SHA_LINE"
    fi
fi

# Patch 4: install /opt/livos/livos-rollback.sh
if [[ -f "$ROLLBACK_SH_PATH" ]] && grep -qF "$MARKER_ROLLBACK" "$ROLLBACK_SH_PATH"; then
    echo "ALREADY-PATCHED (REL-02 rollback.sh): $ROLLBACK_SH_PATH present"
else
    cat <<'EOROLLBACK' > "$ROLLBACK_SH_PATH"
<<<INLINE: livos-rollback.sh — full content verbatim including shebang>>>
EOROLLBACK
    chmod 0755 "$ROLLBACK_SH_PATH"
    chown root:root "$ROLLBACK_SH_PATH" 2>/dev/null || true
    echo "PATCH-OK (REL-02 rollback.sh): wrote $ROLLBACK_SH_PATH (mode 0755)"
    if ! bash -n "$ROLLBACK_SH_PATH"; then
        echo "ERROR: $ROLLBACK_SH_PATH failed bash -n — removing" >&2
        rm -f "$ROLLBACK_SH_PATH"
        exit 1
    fi
fi

# Patch 5: install /etc/systemd/system/livos-rollback.service
if [[ -f "$ROLLBACK_UNIT" ]] && grep -qF "$MARKER_UNIT" "$ROLLBACK_UNIT"; then
    echo "ALREADY-PATCHED (REL-02 unit): $ROLLBACK_UNIT present"
else
    cat <<'EOUNIT' > "$ROLLBACK_UNIT"
<<<INLINE: livos-rollback.service — full content verbatim>>>
EOUNIT
    chmod 0644 "$ROLLBACK_UNIT"
    echo "PATCH-OK (REL-02 unit): wrote $ROLLBACK_UNIT"
fi

# Patch 6: install drop-in (skipped if systemd < 254)
if [[ "$INSTALL_DROPIN" == "true" ]]; then
    if [[ -f "$DROPIN_PATH" ]] && grep -qF "$MARKER_DROPIN" "$DROPIN_PATH"; then
        echo "ALREADY-PATCHED (REL-02 drop-in): $DROPIN_PATH present"
    else
        mkdir -p "$DROPIN_DIR"
        cat <<'EODROPIN' > "$DROPIN_PATH"
<<<INLINE: auto-rollback.conf — full content verbatim>>>
EODROPIN
        chmod 0644 "$DROPIN_PATH"
        echo "PATCH-OK (REL-02 drop-in): wrote $DROPIN_PATH"
    fi
else
    echo "SKIPPED (REL-02 drop-in): systemd version too old"
fi

systemctl daemon-reload
systemctl enable livos-rollback.service 2>/dev/null || true
echo "INFO: systemctl daemon-reload + livos-rollback.service enabled"

# Final safety net
if ! bash -n "$UPDATE_SH"; then
    echo "FATAL: patched $UPDATE_SH failed syntax — restoring backup" >&2
    cp "$UPDATE_SH.pre-phase32" "$UPDATE_SH"
    chmod +x "$UPDATE_SH"
    echo "RESTORED: $UPDATE_SH reverted to pre-phase32 backup" >&2
    exit 1
fi

echo
echo "=== PHASE 32 PATCH COMPLETE ==="
echo "Backup:           $UPDATE_SH.pre-phase32"
echo "Patched:          $UPDATE_SH"
echo "rollback.sh:      $ROLLBACK_SH_PATH"
echo "rollback unit:    $ROLLBACK_UNIT"
echo "drop-in installed: $INSTALL_DROPIN  (systemd v$SYSTEMD_VER_RAW)"
echo "history dir:      $HISTORY_DIR"
echo
grep -nE "Phase 32 (REL-01|REL-02)" "$UPDATE_SH" | head -10 || true
```

```typescript
export interface ContainerDetail {
  id: string
  name: string
  image: string
  state: string
  status: string
  created: string          // ISO date string from inspect
  platform: string         // e.g. "linux/amd64"
  restartPolicy: string    // e.g. "always", "unless-stopped", "no"
  restartCount: number
  healthStatus: string | null  // "healthy" | "unhealthy" | "starting" | null (no healthcheck)
  ports: PortMapping[]
  volumes: VolumeMount[]
  envVars: string[]        // ["KEY=value", ...] from Config.Env
  networks: string[]       // network names
  mounts: MountInfo[]      // bind mounts and volume mounts with full detail
}

export interface VolumeMount {
  type: string             // "volume" | "bind" | "tmpfs"
  source: string           // host path or volume name
  destination: string      // container path
  readOnly: boolean
}

export interface MountInfo {
  type: string
  source: string
  destination: string
  mode: string             // "rw" | "ro"
}

export interface ContainerStats {
  cpuPercent: number       // 0-100
  memoryUsage: number      // bytes
  memoryLimit: number      // bytes
  memoryPercent: number    // 0-100
  networkRx: number        // bytes received
  networkTx: number        // bytes transmitted
  pids: number             // number of processes
}
```

**docker.ts** -- Add three new exported async functions after the existing `manageContainer` function. Reuse the singleton `docker` instance:

1. `inspectContainer(name: string): Promise<ContainerDetail>` -- Uses `docker.getContainer(name).inspect()`. Maps the raw inspect response to the ContainerDetail type. For ports, reuse the same PortMapping mapping as listContainers. For healthStatus, read `info.State.Health?.Status ?? null`. For platform, read `info.Platform || 'unknown'`. For envVars, read `info.Config.Env || []`. For networks, read `Object.keys(info.NetworkSettings.Networks || {})`. For volumes, map `info.Mounts` to VolumeMount[]. For mounts, map `info.Mounts` to MountInfo[]. Wrap in try/catch -- if container not found, throw `new Error('[not-found] Container not found: ' + name)`.

2. `getContainerLogs(name: string, tail: number = 500, timestamps: boolean = true): Promise<string>` -- Uses `docker.getContainer(name).logs({stdout: true, stderr: true, tail, timestamps, follow: false})`. The result is a Buffer. Docker multiplexed stream format prepends 8-byte headers to each frame (first byte: 1=stdout, 2=stderr; bytes 4-7: frame length). Strip these 8-byte headers from each frame before converting to string. Use `strip-ansi` (already in livinityd deps) to strip ANSI codes: `import stripAnsi from 'strip-ansi'`. Apply stripAnsi to the final string. Return the cleaned string. Wrap in try/catch for not-found.

3. `getContainerStats(name: string): Promise<ContainerStats>` -- Uses `docker.getContainer(name).stats({stream: false})`. Calculates cpuPercent as: `cpuDelta = stats.cpu_stats.cpu_usage.total_usage - stats.precpu_stats.cpu_usage.total_usage; systemDelta = stats.cpu_stats.system_cpu_usage - stats.precpu_stats.system_cpu_usage; numCpus = stats.cpu_stats.online_cpus || stats.cpu_stats.cpu_usage.percpu_usage?.length || 1; cpuPercent = systemDelta > 0 ? (cpuDelta / systemDelta) * numCpus * 100 : 0`. Round cpuPercent to 2 decimals. For memory: `usage = stats.memory_stats.usage - (stats.memory_stats.stats?.cache || 0); limit = stats.memory_stats.limit; memoryPercent = limit > 0 ? (usage / limit) * 100 : 0`. Round memoryPercent to 2 decimals. For network: sum `rx_bytes` and `tx_bytes` across all interfaces in `stats.networks || {}`. For pids: `stats.pids_stats?.current || 0`. Wrap in try/catch for not-found.

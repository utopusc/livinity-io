**use-container-detail.ts** -- Create a hook that wraps three tRPC queries for a single container:

```typescript
import {trpcReact} from '@/trpc/trpc'

export function useContainerDetail(containerName: string | null) {
  // Only enable queries when containerName is non-null (Sheet is open)
  const enabled = containerName !== null

  const inspectQuery = trpcReact.docker.inspectContainer.useQuery(
    {name: containerName!},
    {enabled, retry: false}
  )

  const logsQuery = trpcReact.docker.containerLogs.useQuery(
    {name: containerName!, tail: 500, timestamps: true},
    {enabled, retry: false, refetchInterval: 2000}  // Poll every 2s per CONTEXT.md
  )

  const statsQuery = trpcReact.docker.containerStats.useQuery(
    {name: containerName!},
    {enabled, retry: false, refetchInterval: 3000}  // Poll every 3s per CONTEXT.md
  )

  return {
    detail: inspectQuery.data ?? null,
    detailLoading: inspectQuery.isLoading && enabled,
    detailError: inspectQuery.error,

    logs: logsQuery.data ?? '',
    logsLoading: logsQuery.isLoading && enabled,
    logsError: logsQuery.error,
    refetchLogs: logsQuery.refetch,

    stats: statsQuery.data ?? null,
    statsLoading: statsQuery.isLoading && enabled,
    statsError: statsQuery.error,
  }
}
```

The hook accepts `containerName: string | null`. When null, all queries are disabled (Sheet closed). Polling starts automatically when a container is selected.

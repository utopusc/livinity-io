**File 1: `streaming-message.tsx`** -- Streaming markdown renderer

Create a React component that wraps the `Streamdown` component for rendering assistant messages:

```tsx
import 'streamdown/styles.css'
import {Streamdown} from 'streamdown'
import {code} from '@streamdown/code'

interface StreamingMessageProps {
  content: string
  isStreaming?: boolean
}

export function StreamingMessage({content, isStreaming = false}: StreamingMessageProps) {
  // Render streamdown with:
  // - plugins={{ code }} for Shiki syntax highlighting
  // - isAnimating={isStreaming} to signal active streaming state
  // - animated prop for smooth token-by-token rendering
  // - children = content string

  // If content is empty and streaming, show a subtle pulsing cursor/dot
  // If content is empty and not streaming, render nothing

  return (
    <div className="prose prose-sm prose-invert max-w-none [&_pre]:bg-zinc-900 [&_pre]:border [&_pre]:border-zinc-700/50 [&_pre]:rounded-lg [&_code]:text-sm">
      {content ? (
        <Streamdown
          animated
          isAnimating={isStreaming}
          plugins={{code}}
        >
          {content}
        </Streamdown>
      ) : isStreaming ? (
        <span className="inline-block h-4 w-1 animate-pulse bg-violet-400 rounded-full" />
      ) : null}
    </div>
  )
}
```

The prose-invert class matches the dark theme. The [&_pre] overrides ensure code blocks have proper dark backgrounds consistent with LivOS design.

```tsx
interface ChatInputProps {
  value: string
  onChange: (value: string) => void
  onSend: () => void
  onStop: () => void
  isStreaming: boolean
  isConnected: boolean
  disabled?: boolean
}
```

Implementation details:

1. **Auto-resize textarea:**
   - `useRef<HTMLTextAreaElement>` for the textarea element
   - On every `onChange`, reset `style.height = 'auto'` then set `style.height = Math.min(scrollHeight, 144) + 'px'` (144px = ~6 lines)
   - `rows={1}` initial, resize-none class
   - Focus the textarea on mount via `useEffect` with `ref.current?.focus()`

2. **Keyboard handling:**
   - `Enter` (without Shift) calls `onSend()` and prevents default
   - `Shift+Enter` inserts newline (default behavior, no prevention)
   - When `isStreaming` is true, Enter does nothing (textarea is disabled)

    export const installHistory = pgTable('install_history', {
      id: uuid('id').defaultRandom().primaryKey(),
      user_id: uuid('user_id').notNull(),
      app_id: uuid('app_id').notNull().references(() => apps.id),
      action: text('action').notNull(),  // 'install' or 'uninstall'

      created_at: timestamp('created_at', { withTimezone: true }).defaultNow().notNull(),
    });

    2. Create the SQL migration file `platform/web/src/db/migrations/0002_create_install_history.sql`:

    ```sql
    CREATE TABLE IF NOT EXISTS install_history (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      user_id UUID NOT NULL,
      app_id UUID NOT NULL REFERENCES apps(id),
      action TEXT NOT NULL CHECK (action IN ('install', 'uninstall')),
      instance_name TEXT NOT NULL,
      created_at TIMESTAMPTZ NOT NULL DEFAULT now()
    );

    CREATE INDEX IF NOT EXISTS idx_install_history_user ON install_history(user_id);
    CREATE INDEX IF NOT EXISTS idx_install_history_app ON install_history(app_id);
    CREATE INDEX IF NOT EXISTS idx_install_history_instance ON install_history(user_id, instance_name);
    ```

    The CHECK constraint ensures only 'install' or 'uninstall' values are stored. Indexes cover the common query patterns: by user, by app, and by user+instance.

    Create 3 new Next.js route handlers following the exact same patterns as the restored app routes (validateApiKey auth, Drizzle ORM queries, NextResponse.json returns).

    **1. POST /api/install-event** (`platform/web/src/app/api/install-event/route.ts`) -- per API-01

    ```typescript
    import { NextRequest, NextResponse } from 'next/server';
    import { validateApiKey, unauthorizedResponse } from '@/lib/api-auth';
    import { db } from '@/lib/drizzle';
    import { installHistory } from '@/db/schema';

    export async function POST(req: NextRequest) {
      const auth = await validateApiKey(req);
      if (!auth.valid) return unauthorizedResponse(auth.error);

      const body = await req.json();
      const { app_id, action, instance_name } = body;

      // Validate required fields
      if (!app_id || !action || !instance_name) {
        return NextResponse.json({ error: 'Missing required fields: app_id, action, instance_name' }, { status: 400 });
      }

      // Validate action value
      if (action !== 'install' && action !== 'uninstall') {
        return NextResponse.json({ error: 'action must be "install" or "uninstall"' }, { status: 400 });
      }

      const [row] = await db.insert(installHistory).values({
        user_id: auth.userId,
        app_id,
        action,
        instance_name,
      }).returning();

      return NextResponse.json(row, { status: 201 });
    }
    ```

    Wrap the body parsing and DB insert in a try/catch -- return 400 for JSON parse errors and 500 for DB errors (e.g., invalid app_id UUID reference).

    **2. GET /api/user/apps** (`platform/web/src/app/api/user/apps/route.ts`) -- per API-02

    This endpoint returns the user's currently installed apps grouped by instance. The logic: for each (app_id, instance_name) pair, find the most recent event. If it's 'install', include it; if 'uninstall', exclude it.

    Use a raw SQL query via the `pool` from `@/lib/db` (this is a complex aggregation query that's cleaner in raw SQL than Drizzle):

    ```sql
    WITH latest_events AS (
      SELECT DISTINCT ON (app_id, instance_name)
        app_id, instance_name, action, created_at
      FROM install_history
      WHERE user_id = $1
      ORDER BY app_id, instance_name, created_at DESC
    )
    SELECT le.instance_name, le.created_at as installed_at,
           a.id as app_id, a.name, a.tagline, a.icon_url, a.version, a.category
    FROM latest_events le
    JOIN apps a ON a.id = le.app_id
    WHERE le.action = 'install'
    ORDER BY le.instance_name, a.name
    ```

    Group the results by instance_name in JavaScript and return:
    ```json
    {
      "instances": {
        "my-server": [
          { "app_id": "...", "name": "n8n", "tagline": "...", "icon_url": "...", "version": "...", "category": "...", "installed_at": "..." }
        ],
        "home-lab": [...]
      }
    }
    ```

    **3. GET /api/user/profile** (`platform/web/src/app/api/user/profile/route.ts`) -- per API-03

    Use raw SQL via `pool` to query aggregate data:

    ```sql
    -- Get username
    SELECT email FROM users WHERE id = $1

    -- Get instance count + app count from install_history
    WITH latest_events AS (
      SELECT DISTINCT ON (app_id, instance_name)
        app_id, instance_name, action
      FROM install_history
      WHERE user_id = $1
      ORDER BY app_id, instance_name, created_at DESC
    )
    SELECT
      COUNT(DISTINCT instance_name) as instance_count,
      COUNT(*) FILTER (WHERE action = 'install') as app_count
    FROM latest_events
    ```

    Return:
    ```json
    {
      "email": "user@example.com",
      "instance_count": 2,
      "app_count": 5
    }
    ```

    For all 3 endpoints:
    - Import and call `validateApiKey(req)` first -- return `unauthorizedResponse` on failure
    - Use `auth.userId` (string UUID) as the user identifier for all queries
    - Wrap in try/catch, return 500 with `{ error: 'Internal server error' }` on exceptions
    - Do NOT log sensitive data (API keys, user IDs) in error messages

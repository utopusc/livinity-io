    import type {Request, Response, NextFunction} from 'express'
    import {isMultiUserMode} from '../ai/per-user-claude.js'
    import type Livinityd from '../../index.js'

    /**
     * Allowlist: 127.0.0.1, ::1 (IPv6 loopback), and 172.17.0.0/16 (Docker default bridge).
     * Per D-41-08 — only container-network and loopback traffic accepted; reject everything else 401.
     *
     * IPv4-mapped-IPv6 (::ffff:127.0.0.1) is normalized before comparison.
     *
     * NOTE on threat model: external traffic CAN'T reach this route in production because livinityd's
     * existing CORS + helmet stack handles internet-facing routes elsewhere — but this guard is
     * defense-in-depth in case of future misconfiguration. See 41-AUDIT.md Section 7 for full rationale.
     */
    export function containerSourceIpGuard(req: Request, res: Response, next: NextFunction): void {
      let ip = req.socket.remoteAddress || ''
      // Strip IPv4-mapped-IPv6 prefix
      if (ip.startsWith('::ffff:')) ip = ip.slice(7)
      if (ip === '127.0.0.1' || ip === '::1') return next()
      // CIDR check for 172.17.0.0/16 (default Docker bridge)
      if (ip.startsWith('172.17.')) {
        const parts = ip.split('.')
        if (parts.length === 4 && parts.every((p) => /^\d+$/.test(p) && +p >= 0 && +p <= 255)) {
          return next()
        }
      }
      res.status(401).json({
        type: 'error',
        error: {type: 'authentication_error', message: `request source ip ${ip} not on broker allowlist`},
      })
    }

    /**
     * Resolve userId from URL param + authorize:
     * - 404 if userId is not in the users table
     * - 403 if multi_user_mode is OFF and userId != admin's id (single-user mode = only the admin user has a "container")
     * Returns the resolved user object on success, or undefined after writing the error response.
     */
    export async function resolveAndAuthorizeUserId(
      req: Request,
      res: Response,
      livinityd: Livinityd,
    ): Promise<{userId: string} | undefined> {
      const userId = req.params.userId
      // Defensive userId shape — same regex as getUserClaudeDir; reject path traversal at the gate
      if (!userId || !/^[a-zA-Z0-9_-]+$/.test(userId)) {
        res.status(400).json({type: 'error', error: {type: 'invalid_request_error', message: 'invalid user id'}})
        return undefined
      }

      // VERIFY users-table lookup API during execution: livinityd.users.getById? livinityd.users.find?
      // 41-AUDIT.md Section 4 should document the exact method; if it does not, audit livinityd.users module
      // (livos/packages/livinityd/source/modules/users/) to find the right call before implementing.
      const user = await livinityd.users.getById(userId).catch(() => null)
      if (!user) {
        res.status(404).json({type: 'error', error: {type: 'not_found_error', message: 'user not found'}})
        return undefined
      }

      const multiUser = await isMultiUserMode(livinityd)
      if (!multiUser) {
        // Single-user mode: only the admin user is permitted (the broker is a multi-user feature;
        // single-user mode users should call /api/agent/stream directly via the existing chat UI).
        const admin = await livinityd.users.getAdmin().catch(() => null)
        if (!admin || admin.id !== userId) {
          res.status(403).json({
            type: 'error',
            error: {type: 'forbidden', message: 'single-user mode: only admin user permitted on broker'},
          })
          return undefined
        }
      }

      return {userId}
    }

    **translate-request.ts** — pure-function translator (no I/O, no Express deps; easy to unit-test):
    ```typescript
    import type {AnthropicMessagesRequest} from './types.js'

    export interface SdkRunArgs {
      task: string
      contextPrefix?: string
      systemPromptOverride?: string
    }

    /**
     * Translate an Anthropic Messages request into SdkAgentRunner.run() arguments.
     *
     * Strategy (per D-41-15):
     * - The LATEST user message becomes `task`
     * - All PRIOR messages (any role) are formatted as `Previous conversation:\n<role>: <text>\n\n...`
     *   and become `contextPrefix` (matches existing AI Chat pattern at livinityd ai/index.ts:433)
     * - The `system` field (string or content blocks) becomes `systemPromptOverride`
     * - Client-provided `tools` array is IGNORED per D-41-14; caller MUST log a warning when present
     *
     * Throws Error with descriptive message on invalid input shape.
     */
    export function translateAnthropicMessagesToSdkArgs(req: AnthropicMessagesRequest): SdkRunArgs {
      if (!req || typeof req !== 'object') throw new Error('request must be an object')
      if (!Array.isArray(req.messages) || req.messages.length === 0) {
        throw new Error('messages must be a non-empty array')
      }
      // Find latest user message
      const latestUserIdx = [...req.messages].reverse().findIndex((m) => m?.role === 'user')
      if (latestUserIdx === -1) throw new Error('no user message found in messages array')
      const lastIdx = req.messages.length - 1 - latestUserIdx
      const lastUserMsg = req.messages[lastIdx]
      const task = extractText(lastUserMsg)

      // Prior turns (everything before the latest user message)
      const priorTurns = req.messages.slice(0, lastIdx)
      let contextPrefix: string | undefined
      if (priorTurns.length > 0) {
        const formatted = priorTurns
          .map((m) => `${m.role === 'user' ? 'User' : 'Assistant'}: ${extractText(m)}`)
          .join('\n\n')
        contextPrefix = `Previous conversation:\n${formatted}`
      }

      // System prompt
      let systemPromptOverride: string | undefined
      if (typeof req.system === 'string') {
        systemPromptOverride = req.system
      } else if (Array.isArray(req.system)) {
        systemPromptOverride = req.system.filter((b) => b?.type === 'text').map((b) => b.text).join('\n')
      }

      return {task, contextPrefix, systemPromptOverride}
    }

    function extractText(msg: {content: string | Array<{type: string; text?: string}>}): string {
      if (typeof msg.content === 'string') return msg.content
      if (Array.isArray(msg.content)) {
        return msg.content.filter((b) => b?.type === 'text' && typeof b.text === 'string').map((b) => b.text!).join('\n')
      }
      throw new Error('message content must be string or text-block array')
    }
    ```

    **router.ts** — Express Router with full middleware chain (IP guard → user auth → body validation → stub handler):
    ```typescript
    import express, {type Request, type Response} from 'express'
    import {containerSourceIpGuard, resolveAndAuthorizeUserId} from './auth.js'
    import {translateAnthropicMessagesToSdkArgs} from './translate-request.js'
    import type {AnthropicMessagesRequest, BrokerDeps} from './types.js'

    /**
     * Create the broker Express router.
     *
     * Route surface (Plan 41-02):
     *   POST /:userId/v1/messages — accepts Anthropic Messages API body, returns stub JSON.
     *
     * Middleware chain (in order):
     *   1. containerSourceIpGuard — reject non-loopback / non-Docker-bridge IPs (401)
     *   2. express.json({limit: '10mb'}) — parse JSON body (note: Anthropic Messages bodies can include images later; 10mb headroom)
     *   3. resolveAndAuthorizeUserId — 404/403 unauthorized userIds
     *   4. body validation — 400 for invalid Anthropic Messages shape
     *   5. handler — Plan 41-02 stub; Plan 41-03 wires SdkAgentRunner
     *
     * NOTE: Plan 41-02 returns a stub 200 JSON. Plan 41-03 replaces the stub
     * with the real SdkAgentRunner-backed sync + SSE responses.
     */
    export function createBrokerRouter(deps: BrokerDeps): express.Router {
      const router = express.Router()

      // 1. IP guard — first thing, before body parsing
      router.use(containerSourceIpGuard)

      // 2. JSON body parsing (10MB limit — leaves room for future image content blocks)
      router.use(express.json({limit: '10mb'}))

      // 3 + 4 + 5. POST /:userId/v1/messages
      router.post('/:userId/v1/messages', async (req: Request, res: Response) => {
        // 3. user_id resolution + authorization
        const auth = await resolveAndAuthorizeUserId(req, res, deps.livinityd)
        if (!auth) return  // response already written

        // 4. body validation
        const body = req.body as AnthropicMessagesRequest
        if (!body || typeof body !== 'object') {
          res.status(400).json({type: 'error', error: {type: 'invalid_request_error', message: 'request body must be JSON'}})
          return
        }
        if (typeof body.model !== 'string' || body.model.length === 0) {
          res.status(400).json({type: 'error', error: {type: 'invalid_request_error', message: 'model must be a non-empty string'}})
          return
        }
        if (!Array.isArray(body.messages) || body.messages.length === 0) {
          res.status(400).json({type: 'error', error: {type: 'invalid_request_error', message: 'messages must be a non-empty array'}})
          return
        }

        // Per D-41-14: client-provided tools are ignored with warn log
        if (Array.isArray(body.tools) && body.tools.length > 0) {
          deps.livinityd.logger.log(
            `[livinity-broker] WARN client provided ${body.tools.length} tools — ignoring per D-41-14 (broker only exposes LivOS MCP tools)`,
          )
        }

        // Translate (validates message shape; throws on invalid)
        let sdkArgs
        try {
          sdkArgs = translateAnthropicMessagesToSdkArgs(body)
        } catch (err: any) {
          res.status(400).json({
            type: 'error',
            error: {type: 'invalid_request_error', message: err?.message || 'translation failed'},
          })
          return
        }

        // 5. Plan 41-02 STUB — Plan 41-03 wires SdkAgentRunner here.
        // Returning the parsed translation as JSON allows Plan 41-02 integration tests to verify
        // the pre-handler chain end-to-end without needing a real SdkAgentRunner mock.
        res.status(200).json({
          stub: true,
          phase: '41-02',
          userId: auth.userId,
          stream: body.stream === true,
          translated: sdkArgs,
          notice: 'Plan 41-03 will replace this stub with the real SdkAgentRunner-backed response.',
        })
      })

      return router
    }
    ```

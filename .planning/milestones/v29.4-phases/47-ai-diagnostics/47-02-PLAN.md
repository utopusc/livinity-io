    export function makeFlushAndResync(deps: {
      redis: RedisLike
      pg: PgPoolLike
      syncAll: () => Promise<void>  // injected — calls into nexus/core CapabilityRegistry.syncAll
      redisPrefix?: string
      auditWriter?: (row: AuditRow) => Promise<string>  // returns inserted row id
    }) {
      const PREFIX = deps.redisPrefix ?? 'nexus:cap:'
      const PENDING_PREFIX = `${PREFIX}_pending:`

      return {
        async run(opts: {scope: FlushScope; actorUserId?: string}): Promise<FlushAndResyncResult> {
          const start = Date.now()

          // Phase 1: Read user overrides BEFORE flush (B-07 mitigation).
          // Feature-flag via to_regclass — table may not exist yet (G-03).
          const overridesPreserved: string[] = []
          try {
            const exists = await deps.pg.query(
              `SELECT to_regclass('public.user_capability_overrides') AS r`
            )
            const tableExists = (exists.rows[0] as {r: string|null})?.r != null
            if (tableExists) {
              const rows = await deps.pg.query(
                `SELECT capability_id FROM user_capability_overrides WHERE enabled = false`
              )
              for (const r of rows.rows as {capability_id: string}[]) {
                overridesPreserved.push(r.capability_id)
              }
            } else {
              // Non-fatal: log + skip
              console.warn('[diagnostics] user_capability_overrides table missing — skipping override re-apply')
            }
          } catch (err) {
            // Non-fatal: continue with empty override list
          }

          // Phase 2: Count BEFORE
          const beforeKeys = await scopedKeys(deps.redis, PREFIX, opts.scope)
          const before = beforeKeys.length

          // Phase 3: Build PENDING via syncAll() but redirected to PENDING_PREFIX.
          // OPTION (cleanest): override CapabilityRegistry.REDIS_PREFIX via DI OR
          // call syncAll() into a scratch Redis, then mass-RENAME under Lua.
          // Simpler approach: syncAll() writes to PREFIX as usual, but wrap in a transaction.
          // TRADE-OFF: the cleanest atomic-swap requires syncAll() accepting a prefix arg.
          // For Phase 47, use the following Lua script for the SWAP (post-syncAll):
          //   - The syncAll writes to a "_pending" mirror prefix via a thin redis-wrapper that intercepts SET calls.
          //   - We provide an injectable redis wrapper that translates `${PREFIX}<id>` → `${PENDING_PREFIX}<id>` for writes ONLY during this flush.
          // (Implementer: see "Inject prefixed-redis-write-wrapper" pattern below.)

          await deps.syncAll()  // The injected syncAll uses the wrapper that writes to PENDING_PREFIX

          // Phase 4: Atomic Lua swap.
          // The Lua script:
          //   - SCANs PENDING_PREFIX*
          //   - For each pending key, RENAMEs to live prefix (overwriting)
          //   - DELs any live keys NOT covered by the rename (stale entries)
          //   - PRESERVES `${PREFIX}_meta:*` and `${PREFIX}_audit*` (W-14)
          // (Lua source embedded as a string constant — see ATOMIC_SWAP_LUA below.)
          await deps.redis.eval(ATOMIC_SWAP_LUA, 0, PREFIX, PENDING_PREFIX, opts.scope)

          // Phase 5: Re-apply user overrides (B-07).
          for (const capId of overridesPreserved) {
            // Read the just-swapped manifest, mutate enabled=false, write back.
            const key = `${PREFIX}${capId}`
            const raw = await deps.redis.get(key)
            if (raw) {
              const manifest = JSON.parse(raw) as {enabled?: boolean}
              manifest.enabled = false
              const pl = deps.redis.pipeline()
              pl.set(key, JSON.stringify(manifest))
              await pl.exec()
            }
          }

          // Phase 6: Count AFTER + write audit
          const afterKeys = await scopedKeys(deps.redis, PREFIX, opts.scope)
          const after = afterKeys.length
          const durationMs = Date.now() - start

          // Audit: Redis history list (LTRIM 99)

            ts: new Date().toISOString(),
            actor: opts.actorUserId ?? null,
            scope: opts.scope,
            before, after,
            overridesPreserved,
            durationMs,

    // ---- Helpers ----

    async function scopedKeys(redis: RedisLike, prefix: string, scope: FlushScope): Promise<string[]> {
      // For 'builtins': only keys matching ${prefix}tool:* (the built-in scope)
      // For 'all': keys matching ${prefix}{tool|skill|mcp|hook|agent}:* but NOT _meta/_audit*

        const keys = await redis.keys(`${prefix}${t}:*`)
        for (const k of keys) out.push(k)
      }
      return out

    function defaultPreconditionEvaluator(redis: RedisLike): PreconditionEvaluator {
      return async (capabilityId) => {

          return {met: !!process.env.SERPER_API_KEY, reason: process.env.SERPER_API_KEY ? undefined : 'SERPER_API_KEY not set'}

          return {met: flag === 'true', reason: flag === 'true' ? undefined : `${service} not connected`}
        }
        return {met: true}  // shell, docker_*, files_* have no precondition

    // The atomic-swap Lua script as a string constant.
    // ARGV[1] = live prefix (e.g. 'nexus:cap:')
    // ARGV[2] = pending prefix (e.g. 'nexus:cap:_pending:')
    // ARGV[3] = scope ('builtins' | 'all')
    // The script SCANs pending, RENAMEs each to live, deletes stale lives that match the scope.
    const ATOMIC_SWAP_LUA = `
      local livePrefix = ARGV[1]
      local pendingPrefix = ARGV[2]
      local scope = ARGV[3]

      -- Determine which manifest types are in scope
      local types
      if scope == 'builtins' then
        types = {'tool'}
      else
        types = {'tool', 'skill', 'mcp', 'hook', 'agent'}
      end

      -- For each type, collect pending keys, RENAME each to live (overwriting).
      -- Then collect live keys for that type and DEL any not in the rename set.
      local renamedSet = {}
      for _, t in ipairs(types) do
        local pendingKeys = redis.call('KEYS', pendingPrefix .. t .. ':*')
        for _, pk in ipairs(pendingKeys) do
          -- pk = "nexus:cap:_pending:tool:shell"; live = "nexus:cap:tool:shell"
          local suffix = string.sub(pk, string.len(pendingPrefix) + 1)
          local lk = livePrefix .. suffix
          redis.call('RENAME', pk, lk)
          renamedSet[lk] = true
        end

        local liveKeys = redis.call('KEYS', livePrefix .. t .. ':*')
        for _, lk in ipairs(liveKeys) do
          if not renamedSet[lk] then
            redis.call('DEL', lk)
          end
        end
      end

      -- Update meta last_sync_at
      redis.call('SET', livePrefix .. '_meta:last_sync_at', ARGV[4] or tostring(redis.call('TIME')[1]))

      return 'OK'
    `

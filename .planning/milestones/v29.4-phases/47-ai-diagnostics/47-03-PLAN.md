    // Types
    export type ModelIdentityVerdict = 'clean' | 'dist-drift' | 'source-confabulation' | 'both' | 'inconclusive'

    export interface DiagnoseModelIdentityResult {
      verdict: ModelIdentityVerdict
      steps: {
        step1_brokerProbe: {url: string; responseModel: string | null; expected: string; match: boolean; error?: string}
        step2_responseModelInterpretation: string
        step3_environSnapshot: {pids: number[]; envs: Record<string, string>[]} | 'NONE'
        step4_pnpmStoreCount: {dirs: string[]; count: number; driftRisk: boolean}
        step5_resolvedDist: {path: string; ls: string}
        step6_identityMarker: {resolvedDistPath: string; markerCount: number; mtime: number; markerPresent: boolean}
      }
      capturedAt: string  // ISO
    }

    export type ExecFileFn = (binary: string, args: string[], opts: {timeout: number})
      => Promise<{stdout: string; stderr: string}>

    export type FetchFn = typeof fetch

    export interface ModelIdentityDeps {
      execFile: ExecFileFn
      fetch: FetchFn
      brokerBaseUrl?: string  // default 'http://localhost:8080'
      adminUserIdResolver?: () => Promise<string | null>  // resolves admin userId for broker URL
      expectedModelId?: string  // default 'claude-opus-4-7' (per tierToModel 43.12)
    }

    // Constants — D-NO-SERVER4 hard-wall (G-10)
    const FORBIDDEN_HOSTS = ['45.137.194.103', '45.137.194.102']

    export function makeDiagnoseModelIdentity(deps: ModelIdentityDeps) {
      return {
        async diagnose(): Promise<DiagnoseModelIdentityResult> {
          // Refuse if brokerBaseUrl looks like Server4/5
          const baseUrl = deps.brokerBaseUrl ?? 'http://localhost:8080'
          for (const h of FORBIDDEN_HOSTS) {
            if (baseUrl.includes(h)) {
              throw new Error(`D-NO-SERVER4: refusing to diagnose against forbidden host ${h}`)
            }
          }

          const expected = deps.expectedModelId ?? 'claude-opus-4-7'

          // Step 1: broker probe
          let step1: DiagnoseModelIdentityResult['steps']['step1_brokerProbe']
          try {
            const userId = (await deps.adminUserIdResolver?.()) ?? 'admin'
            const url = `${baseUrl}/u/${userId}/v1/messages`
            const ctl = new AbortController()
            const t = setTimeout(() => ctl.abort(), 10_000)
            try {
              const r = await deps.fetch(url, {
                method: 'POST',
                headers: {'content-type': 'application/json', 'anthropic-version': '2023-06-01'},
                body: JSON.stringify({
                  model: expected,
                  max_tokens: 50,
                  messages: [{role: 'user', content: 'What model are you? Respond with only the exact model ID.'}],
                }),
                signal: ctl.signal,
              })
              const json = await r.json() as {model?: string}
              const responseModel = json?.model ?? null
              step1 = {url, responseModel, expected, match: responseModel === expected}
            } finally { clearTimeout(t) }
          } catch (err: any) {
            step1 = {url: baseUrl, responseModel: null, expected, match: false, error: err?.message ?? 'fetch_failed'}
          }

          const step2 = step1.match
            ? `response.model literal "${step1.responseModel}" matches expected — broker layer clean`
            : `response.model literal "${step1.responseModel}" disagrees with expected "${expected}" — possible source-confabulation`

          // Step 3: /proc/<claude-pid>/environ
          let step3: DiagnoseModelIdentityResult['steps']['step3_environSnapshot']
          try {
            const pgrepR = await deps.execFile('pgrep', ['-af', 'claude'], {timeout: 5_000})
            const pids = pgrepR.stdout.split('\n').map((l) => parseInt(l.trim().split(' ')[0])).filter((n) => !isNaN(n))
            if (pids.length === 0) {
              step3 = 'NONE'
            } else {
              const envs: Record<string, string>[] = []
              for (const pid of pids) {
                try {
                  const er = await deps.execFile('cat', [`/proc/${pid}/environ`], {timeout: 2_000})
                  const env: Record<string, string> = {}
                  for (const kv of er.stdout.split('\0')) {
                    if (/^(ANTHROPIC|CLAUDE|HOME|PATH)=/.test(kv)) {
                      const eq = kv.indexOf('=')
                      env[kv.slice(0, eq)] = kv.slice(eq + 1)
                    }
                  }
                  envs.push(env)
                } catch { /* swallow ENOENT */ }
              }
              step3 = {pids, envs}
            }
          } catch {
            step3 = 'NONE'
          }

          // Step 4: pnpm-store dir count
          let step4: DiagnoseModelIdentityResult['steps']['step4_pnpmStoreCount']
          try {
            const lsR = await deps.execFile('ls', ['-la', '/opt/livos/node_modules/.pnpm/'], {timeout: 5_000})
            const dirs = lsR.stdout.split('\n').filter((l) => /@nexus\+core/.test(l))
            step4 = {dirs, count: dirs.length, driftRisk: dirs.length > 1}
          } catch (err: any) {
            step4 = {dirs: [], count: 0, driftRisk: false}
          }

          // Step 5: readlink -f resolved
          let step5: DiagnoseModelIdentityResult['steps']['step5_resolvedDist']
          try {
            const rlR = await deps.execFile('readlink', ['-f', '/opt/livos/node_modules/@nexus/core'], {timeout: 5_000})
            const lsR = await deps.execFile('ls', ['-la', '/opt/livos/node_modules/@nexus/core'], {timeout: 5_000})
            step5 = {path: rlR.stdout.trim(), ls: lsR.stdout}
          } catch {
            step5 = {path: '', ls: ''}
          }

          // Step 6: marker grep + mtime
          let step6: DiagnoseModelIdentityResult['steps']['step6_identityMarker']
          try {
            const distPath = `${step5.path}/dist/index.js`
            const grepR = await deps.execFile('grep', ['-c', 'You are powered by the model named', distPath], {timeout: 5_000})
            const markerCount = parseInt(grepR.stdout.trim()) || 0
            const statR = await deps.execFile('stat', ['-c', '%Y', distPath], {timeout: 5_000})
            const mtime = parseInt(statR.stdout.trim()) || 0
            step6 = {resolvedDistPath: distPath, markerCount, mtime, markerPresent: markerCount > 0}
          } catch {
            step6 = {resolvedDistPath: '', markerCount: 0, mtime: 0, markerPresent: false}
          }

          // Verdict computation per Pattern Map
          let verdict: ModelIdentityVerdict
          if (step1.error) {
            verdict = 'inconclusive'
          } else if (step4.driftRisk && !step6.markerPresent) {
            verdict = step1.match ? 'dist-drift' : 'both'
          } else if (step6.markerPresent && !step1.match) {
            verdict = 'source-confabulation'
          } else if (step1.match && step6.markerPresent && !step4.driftRisk) {
            verdict = 'clean'
          } else {
            verdict = 'inconclusive'
          }

          return {
            verdict,
            steps: {
              step1_brokerProbe: step1,
              step2_responseModelInterpretation: step2,
              step3_environSnapshot: step3,
              step4_pnpmStoreCount: step4,
              step5_resolvedDist: step5,
              step6_identityMarker: step6,
            },
            capturedAt: new Date().toISOString(),
          }
        },
      }
    }

    // Production-wired
    const execFileP = promisify(execFileCb)
    const realExec: ExecFileFn = async (binary, args, opts) => {
      const r = await execFileP(binary, args, {timeout: opts.timeout, encoding: 'utf8'})
      return {stdout: r.stdout as string, stderr: r.stderr as string}
    }

    export const realDiagnoseModelIdentity = makeDiagnoseModelIdentity({
      execFile: realExec,
      fetch: globalThis.fetch,
      // adminUserIdResolver wires from existing user/db helpers — verify import path
    })

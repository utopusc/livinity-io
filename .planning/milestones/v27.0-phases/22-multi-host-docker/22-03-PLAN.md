```typescript
class DeviceBridge {
  private pendingRequests = new Map<string, PendingRequest>()
  // Pattern: outgoing message has requestId; incoming response carries same requestId; resolve the matching pending promise; timeout after 30s
  async executeOnDevice(...): Promise<Result> {
    const requestId = randomUUID()
    this.sendTunnelMessage({type, requestId, ...})
    return new Promise((resolve) => {
      const timeout = setTimeout(() => resolve({error: 'timeout'}), 30000)
      this.pendingRequests.set(requestId, {resolve, timeout})
    })
  }
  onToolResult(event: {requestId, result}): void {
    const pending = this.pendingRequests.get(event.requestId)
    if (!pending) return
    clearTimeout(pending.timeout)
    pending.resolve(event.result)
  }
}
```

    ```typescript
    import {createHash, randomBytes} from 'node:crypto'
    import {query} from '../database/index.js'

    export interface DockerAgentRow {
      id: string
      envId: string
      tokenHash: string
      createdBy: string | null
      createdAt: Date
      lastSeen: Date | null
      revokedAt: Date | null
    }

    function hashToken(token: string): string {
      return createHash('sha256').update(token, 'utf-8').digest('hex')
    }

    export async function createAgent(opts: {envId: string; createdBy: string | null}): Promise<{agent: DockerAgentRow; token: string}> {
      const token = randomBytes(32).toString('hex')  // 64-char hex
      const tokenHash = hashToken(token)
      const row = await query<DockerAgentRow>(
        `INSERT INTO docker_agents (env_id, token_hash, created_by)
         VALUES ($1, $2, $3) RETURNING *`,
        [opts.envId, tokenHash, opts.createdBy],
      )
      // Backfill environments.agent_id so the env can be matched on connect
      await query(`UPDATE environments SET agent_id = $1 WHERE id = $2`, [row[0].id, opts.envId])
      return {agent: row[0], token}
    }

    export async function listAgents(envId?: string): Promise<DockerAgentRow[]> {
      if (envId) {
        return query(`SELECT * FROM docker_agents WHERE env_id = $1 ORDER BY created_at DESC`, [envId])
      }
      return query(`SELECT * FROM docker_agents ORDER BY created_at DESC`, [])
    }

    export async function findAgentByTokenHash(tokenHash: string): Promise<DockerAgentRow | null> {
      const rows = await query<DockerAgentRow>(
        `SELECT * FROM docker_agents WHERE token_hash = $1 AND revoked_at IS NULL`,
        [tokenHash],
      )
      return rows[0] ?? null
    }

    // For verifying a presented cleartext token: hash it, then look up
    export async function findAgentByToken(token: string): Promise<DockerAgentRow | null> {
      return findAgentByTokenHash(hashToken(token))
    }

    export async function revokeAgent(id: string): Promise<void> {
      await query(`UPDATE docker_agents SET revoked_at = NOW() WHERE id = $1 AND revoked_at IS NULL`, [id])
    }

    export async function touchLastSeen(id: string): Promise<void> {
      await query(`UPDATE docker_agents SET last_seen = NOW() WHERE id = $1`, [id])
    }

    export {hashToken}
    ```

    Create `livos/packages/livinityd/source/modules/docker/agent-registry.ts`:

    ```typescript
    import {randomUUID} from 'node:crypto'
    import type {WebSocket} from 'ws'
    import type {AgentRequest, AgentResponse} from './agent-protocol.js'

    interface PendingRequest {
      resolve: (result: unknown) => void
      reject: (err: Error) => void
      timeout: ReturnType<typeof setTimeout>
    }

    interface ConnectedAgent {
      ws: WebSocket
      pending: Map<string, PendingRequest>
    }

    const REQUEST_TIMEOUT_MS = 30_000

    export class AgentRegistry {
      private agents = new Map<string, ConnectedAgent>()

      registerAgent(agentId: string, ws: WebSocket): void {
        // If an existing connection is present, drop it (newer wins)
        const existing = this.agents.get(agentId)
        if (existing) {
          try { existing.ws.close(4409, 'replaced-by-new-connection') } catch {}
          for (const [, pending] of existing.pending) {
            clearTimeout(pending.timeout)
            pending.reject(new Error('[agent-replaced]'))
          }
        }
        this.agents.set(agentId, {ws, pending: new Map()})
      }

      unregisterAgent(agentId: string): void {
        const conn = this.agents.get(agentId)
        if (!conn) return
        for (const [, pending] of conn.pending) {
          clearTimeout(pending.timeout)
          pending.reject(new Error('[agent-disconnected]'))
        }
        this.agents.delete(agentId)
      }

      isAgentOnline(agentId: string): boolean {
        return this.agents.has(agentId)
      }

      async sendRequest(agentId: string, method: string, args: unknown[]): Promise<unknown> {
        const conn = this.agents.get(agentId)
        if (!conn) throw new Error(`[agent-offline] agent ${agentId} is not connected`)
        const requestId = randomUUID()
        const message: AgentRequest = {type: 'request', requestId, method, args}
        return new Promise<unknown>((resolve, reject) => {
          const timeout = setTimeout(() => {
            conn.pending.delete(requestId)
            reject(new Error(`[agent-timeout] ${method} did not respond within ${REQUEST_TIMEOUT_MS}ms`))
          }, REQUEST_TIMEOUT_MS)
          conn.pending.set(requestId, {resolve, reject, timeout})
          try {
            conn.ws.send(JSON.stringify(message))
          } catch (err: any) {
            clearTimeout(timeout)
            conn.pending.delete(requestId)
            reject(new Error(`[agent-send-failed] ${err.message}`))
          }
        })
      }

      handleResponse(agentId: string, response: AgentResponse): void {
        const conn = this.agents.get(agentId)
        if (!conn) return
        const pending = conn.pending.get(response.requestId)
        if (!pending) return
        clearTimeout(pending.timeout)
        conn.pending.delete(response.requestId)
        if (response.error) {
          const err = new Error(response.error.message)
          ;(err as any).statusCode = response.error.statusCode
          ;(err as any).code = response.error.code
          pending.reject(err)
        } else {
          pending.resolve(response.result)
        }
      }

      // Used by Redis pub/sub revocation handler
      forceDisconnect(agentId: string, reason: string): void {
        const conn = this.agents.get(agentId)
        if (!conn) return
        try { conn.ws.close(4403, reason) } catch {}
        // unregisterAgent will be called from the WS close handler in agent-socket.ts
      }
    }

    // Module-level singleton — match docker.ts pattern
    export const agentRegistry = new AgentRegistry()
    ```

    ```typescript
    import os from 'node:os'
    import {WebSocket} from 'ws'

    import {dispatch, getDockerVersion} from './proxy.js'
    import type {AgentMessage, AgentRegister} from './protocol.js'

    const AGENT_VERSION = '0.1.0'

    function parseArgs(): {token: string; server: string} {
      const args = process.argv.slice(2)
      const get = (flag: string): string | undefined => {
        const i = args.indexOf(flag)
        return i >= 0 ? args[i + 1] : undefined
      }
      const token = get('--token') ?? process.env.LIVOS_AGENT_TOKEN
      const server = get('--server') ?? process.env.LIVOS_AGENT_SERVER
      if (!token) throw new Error('--token (or LIVOS_AGENT_TOKEN) required')
      if (!server) throw new Error('--server (or LIVOS_AGENT_SERVER) required, e.g. wss://livinity.cloud/agent/connect')
      return {token, server}
    }

    async function connectLoop({token, server}: {token: string; server: string}): Promise<void> {
      let backoff = 1_000
      const MAX_BACKOFF = 30_000
      const dockerVersion = await getDockerVersion()

      while (true) {
        try {
          await runOnce({token, server, dockerVersion})
          backoff = 1_000  // reset on successful clean disconnect
        } catch (err: any) {
          console.error(`[agent] connection error: ${err.message}`)
        }
        console.log(`[agent] reconnecting in ${backoff / 1000}s`)
        await new Promise((r) => setTimeout(r, backoff))
        backoff = Math.min(MAX_BACKOFF, backoff * 2)
      }
    }

    function runOnce({token, server, dockerVersion}: {token: string; server: string; dockerVersion?: string}): Promise<void> {
      return new Promise((resolve, reject) => {
        const ws = new WebSocket(server)
        let pingTimer: ReturnType<typeof setInterval> | null = null

        ws.on('open', () => {
          const reg: AgentRegister = {
            type: 'register',
            token,
            agentVersion: AGENT_VERSION,
            platform: `${os.platform()}-${os.arch()}`,
            dockerVersion,
          }
          ws.send(JSON.stringify(reg))
        })

        ws.on('message', async (raw) => {
          let msg: AgentMessage
          try {
            msg = JSON.parse(raw.toString()) as AgentMessage
          } catch {
            return  // ignore malformed
          }

          if (msg.type === 'registered') {
            console.log(`[agent] registered as ${msg.agentId}`)
            // Start pong heartbeat in response to server pings
            return
          }

          if (msg.type === 'ping') {
            ws.send(JSON.stringify({type: 'pong', ts: Date.now()}))
            return
          }

          if (msg.type === 'request') {
            const response = await dispatch(msg)
            ws.send(JSON.stringify(response))
            return
          }
        })

        ws.on('close', (code, reasonBuf) => {
          if (pingTimer) clearInterval(pingTimer)
          const reason = reasonBuf?.toString?.() ?? ''
          if (code === 4401) {
            console.error(`[agent] auth failed (code 4401): ${reason}. Token is invalid or revoked. Halting.`)
            process.exit(2)  // Don't reconnect on auth failure
          }
          if (code === 4403) {
            console.error(`[agent] disconnected by server (code 4403): ${reason}. Token revoked. Halting.`)
            process.exit(3)
          }
          console.log(`[agent] WS closed code=${code} reason=${reason}`)
          resolve()  // signal reconnect loop to retry
        })

        ws.on('error', (err) => {
          if (pingTimer) clearInterval(pingTimer)
          reject(err)
        })
      })
    }

    const args = parseArgs()
    connectLoop(args).catch((err) => {
      console.error('[agent] fatal:', err)
      process.exit(1)
    })
    ```
